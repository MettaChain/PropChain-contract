13/// Test scalability with increasing property database size
14#[test]
15fn scalability_test_growing_database() {
16    println!("📊 Starting Scalability Test - Growing Database Size");
17    
18    let accounts = default_accounts::<DefaultEnvironment>();
19    set_caller::<DefaultEnvironment>(accounts.alice);
20    let mut registry = PropertyRegistryContract::new();
21    
22    let test_sizes = vec![100, 500, 1000, 2000];
23    let mut results = Vec::new();
24    
25    for size in &test_sizes {
26        println!("\nTesting with {} properties...", size);
27        
28        // Register properties to reach target size
29        let current_count = registry.property_count();
30        for i in current_count..*size {
31            let metadata = generate_property_metadata(0, i);
32            let _ = registry.register_property(metadata);
33        }
34        
35        // Measure query performance at this scale
36        let start = Instant::now();
37        let mut query_count = 0;
38        
39        for id in 0..*size {
40            let _ = registry.get_property(id as u64);
41            query_count += 1;
42        }
43        
44        let total_time = start.elapsed();
45        let avg_query_time = total_time.as_millis() as f64 / query_count as f64;
46        
47        results.push((*size, avg_query_time));
48        
49        println!(
50            "  Database size: {} | Avg query time: {:.2}ms | Total time: {:?}",
51            size, avg_query_time, total_time
52        );
53    }
54    
55    // Analyze scalability
56    println!("\n📊 SCALABILITY ANALYSIS:");
57    println!("{:<15} {:<20} {:<20}", "Properties", "Avg Query (ms)", "Expected Linear");
58    println!("{}", "-".repeat(55));
59    
60    let base_size = test_sizes[0];
61    let base_time = results[0].1;
62    
63    for (size, avg_time) in &results {
64        let expected_linear = base_time * (*size as f64 / base_size as f64);
65        println!("{:<15} {:<20.2} {:<20.2}", size, avg_time, expected_linear);
66    }
67    
68    // Validate sub-linear or linear scaling
69    let first_time = results[0].1;
70    let last_time = results.last().unwrap().1;
71    let size_ratio = test_sizes.last().unwrap() / test_sizes[0];
72    let time_ratio = last_time / first_time;
73    
74    assert!(
75        time_ratio <= (size_ratio as f64) * 1.5,
76        "Query time scaled worse than linear ({}x time vs {}x data)",
77        time_ratio,
78        size_ratio
79    );
80    
81    println!("✅ System shows acceptable scaling with database growth");
82}

84/// Test concurrent user scalability
85#[test]
86fn scalability_test_concurrent_users() {
87    println!("👥 Starting Scalability Test - Increasing Concurrent Users");
88    
89    let user_counts = vec![5, 10, 20, 40];
90    let mut results = Vec::new();
91    
92    for user_count in &user_counts {
93        println!("\nTesting with {} concurrent users...", user_count);
94        
95        let config = LoadTestConfig {
96            concurrent_users: *user_count,
97            duration_secs: 30,
98            ramp_up_secs: 5,
99            operation_delay_ms: 100,
100            target_ops_per_second: *user_count * 5,
101        };
102        
103        let metrics = run_concurrent_load_test(
104            &config,
105            &format!("Scalability - {} Users", user_count),
106            |user_id, cfg, m| {
107                simulate_user_registration(user_id, 10, cfg, m);
108            },
109        );
110        
111        let ops_sec = *metrics.ops_per_second.lock().unwrap();
112        results.push((*user_count, ops_sec, metrics.avg_response_time_ms()));
113    }
114    
115    // Analyze results
116    println!("\n📊 USER SCALABILITY ANALYSIS:");
117    println!("{:<10} {:<15} {:<20} {:<15}", "Users", "Ops/Sec", "Avg Response (ms)", "Throughput/User");
118    println!("{}", "-".repeat(65));
119    
120    for (users, ops_sec, avg_time) in &results {
121        let throughput_per_user = ops_sec / *users as f64;
122        println!(
123            "{:<10} {:<15.2} {:<20.2} {:<15.2}",
124            users, ops_sec, avg_time, throughput_per_user
125        );
126    }
127    
128    // Check for reasonable throughput scaling
129    let first_result = results[0];
130    let last_result = results.last().unwrap();
131    
132    let user_increase = last_result.0 / first_result.0;
133    let throughput_increase = last_result.1 / first_result.1;
134    
135    // We expect some efficiency loss but not severe degradation
136    assert!(
137        throughput_increase >= (user_increase as f64) * 0.5,
138        "Throughput did not scale adequately with user count",
139    );
140    
141    println!("✅ System demonstrates reasonable user scalability");
142}

144/// Test memory scalability
145#[test]
146fn scalability_test_memory_usage() {
147    println!("💾 Starting Scalability Test - Memory Usage Growth");
148    
149    let accounts = default_accounts::<DefaultEnvironment>();
150    set_caller::<DefaultEnvironment>(accounts.alice);
151    let mut registry = PropertyRegistryContract::new();
152    
153    let batch_sizes = vec![100, 500, 1000, 2000, 3000];
154    let mut memory_data = Vec::new();
155    
156    for batch_size in &batch_sizes {
157        println!("\nRegistering {} properties...", batch_size);
158        
159        let start_mem = 0.0; // Placeholder - would need actual memory measurement
160        
161        for i in 0..*batch_size {
162            let metadata = generate_property_metadata(0, i);
163            let _ = registry.register_property(metadata);
164        }
165        
166        // Note: In Rust tests, we can't easily measure heap memory
167        // In production, use tools like jemalloc-ctl or similar
168        let estimated_mem = *batch_size as f64 * 0.5; // Estimate ~0.5KB per property
169        
170        memory_data.push((*batch_size, estimated_mem));
171        
172        println!(
173            "  Registered: {} | Estimated memory: {:.2} KB",
174            batch_size, estimated_mem
175        );
176    }
177    
178    // Analyze memory growth pattern
179    println!("\n📊 MEMORY GROWTH ANALYSIS:");
180    println!("{:<15} {:<20} {:<20}", "Properties", "Est. Memory (KB)", "KB/Property");
181    println!("{}", "-".repeat(55));
182    
183    for (props, mem) in &memory_data {
184        let kb_per_prop = mem / *props as f64;
185        println!("{:<15} {:<20.2} {:<20.2}", props, mem, kb_per_prop);
186    }
187    
188    // Validate linear memory growth
189    if memory_data.len() >= 2 {
190        let first = memory_data[0];
191        let last = memory_data.last().unwrap();
192        
193        let mem_growth = last.1 / first.1;
194        let data_growth = last.0 / first.0;
195        
196        // Memory should grow roughly linearly with data
197        assert!(
198            (mem_growth - data_growth as f64).abs() < 0.5,
199            "Memory growth deviates significantly from linear"
200        );
201    }
202    
203    println!("✅ Memory usage grows linearly with data size");
204}
