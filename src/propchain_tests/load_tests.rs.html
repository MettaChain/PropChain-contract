1//! Load Testing Framework for PropChain
2//!
3//! This module provides comprehensive load testing capabilities to simulate
4//! high-traffic scenarios and measure system performance under stress.
5//!
6//! # Features
7//!
8//! - **Concurrent User Simulation**: Simulate multiple users performing operations simultaneously
9//! - **Graduated Load Testing**: Gradually increase load to find breaking points
10//! - **Stress Testing**: Push system beyond normal capacity
11//! - **Endurance Testing**: Long-running tests to detect memory leaks and degradation
12//! - **Spike Testing**: Sudden load increases to test system resilience
13//!
14//! # Usage
15//!
16//! ```rust,ignore
17//! // Run concurrent registration test
18//! cargo test --package propchain-tests --test load_tests test_concurrent_property_registration --release
19//!
20//! // Run stress test with custom concurrency
21//! cargo test --package propchain-tests --test load_tests stress_test_mass_registration --release -- --test-threads=10
22//!
23//! // Run endurance test
24//! cargo test --package propchain-tests --test load_tests endurance_test_sustained_load --release -- --test-threads=4
25//! ```

35/// Test configuration for load tests

37pub struct LoadTestConfig {
38    /// Number of concurrent users to simulate
39    pub concurrent_users: usize,
40    /// Duration of the test in seconds
41    pub duration_secs: u64,

48}

108/// Metrics collector for load tests
109#[derive(Debug, Default)]
110pub struct LoadTestMetrics {
111    /// Total operations attempted
112    pub total_operations: Arc<Mutex<u64>>,
113    /// Successful operations
114    pub successful_operations: Arc<Mutex<u64>>,
115    /// Failed operations
116    pub failed_operations: Arc<Mutex<u64>>,
117    /// Total response time in milliseconds
118    pub total_response_time_ms: Arc<Mutex<u128>>,
119    /// Minimum response time in milliseconds
120    pub min_response_time_ms: Arc<Mutex<u128>>,
121    /// Maximum response time in milliseconds
122    pub max_response_time_ms: Arc<Mutex<u128>>,
123    /// Operations per second achieved
124    pub ops_per_second: Arc<Mutex<f64>>,
125    /// Peak memory usage (if available)
126    pub peak_memory_mb: Arc<Mutex<f64>>,
127}
128
129impl LoadTestMetrics {
130    /// Record a successful operation with its response time
131    pub fn record_success(&self, response_time_ms: u128) {
132        *self.total_operations.lock().unwrap() += 1;
133        *self.successful_operations.lock().unwrap() += 1;
134        *self.total_response_time_ms.lock().unwrap() += response_time_ms;
135
136        let mut min = self.min_response_time_ms.lock().unwrap();
137        if *min == 0 || response_time_ms < *min {
138            *min = response_time_ms;
139        }
140
141        let mut max = self.max_response_time_ms.lock().unwrap();
142        if response_time_ms > *max {
143            *max = response_time_ms;
144        }
145    }
146
147    /// Record a failed operation
148    pub fn record_failure(&self) {
149        *self.total_operations.lock().unwrap() += 1;
150        *self.failed_operations.lock().unwrap() += 1;
151    }
152
153    /// Calculate average response time
154    pub fn avg_response_time_ms(&self) -> f64 {
155        let total_ops = *self.successful_operations.lock().unwrap();
156        if total_ops == 0 {
157            return 0.0;
158        }
159        let total_time = *self.total_response_time_ms.lock().unwrap() as f64;
160        total_time / total_ops as f64
161    }
162
163    /// Get success rate percentage
164    pub fn success_rate(&self) -> f64 {
165        let total = *self.total_operations.lock().unwrap();
166        if total == 0 {
167            return 0.0;
168        }
169        let success = *self.successful_operations.lock().unwrap();
170        (success as f64 / total as f64) * 100.0
171    }
172
173    /// Print metrics summary
174    pub fn print_summary(&self, test_name: &str) {
175        println!("\n{}", "=".repeat(80));
176        println!("LOAD TEST RESULTS: {}", test_name);
177        println!("{}", "=".repeat(80));
178        println!(
179            "Total Operations:      {}",
180            *self.total_operations.lock().unwrap()
181        );
182        println!(
183            "Successful:            {} ({:.2}%)",
184            *self.successful_operations.lock().unwrap(),
185            self.success_rate()
186        );
187        println!(
188            "Failed:                {}",
189            *self.failed_operations.lock().unwrap()
190        );
191        println!(
192            "Avg Response Time:     {:.2} ms",
193            self.avg_response_time_ms()
194        );
195        println!(
196            "Min Response Time:     {} ms",
197            *self.min_response_time_ms.lock().unwrap()
198        );
199        println!(
200            "Max Response Time:     {} ms",
201            *self.max_response_time_ms.lock().unwrap()
202        );
203        println!(
204            "Ops/Second:            {:.2}",
205            *self.ops_per_second.lock().unwrap()
206        );
207        println!("{}", "=".repeat(80));
208    }
209}

211/// Helper function to generate test property metadata
212pub fn generate_property_metadata(user_id: usize, property_num: usize) -> PropertyMetadata {
213    PropertyMetadata {
214        location: format!("Property {} by User {}", property_num, user_id),
215        size: (1000 + (property_num * 100)) as u64,
216        legal_description: format!("Legal description for property {}", property_num),
217        valuation: (100_000 + (property_num as u128 * 10_000)),
218        documents_url: format!("ipfs://user{}/prop{}", user_id, property_num),
219    }
220}

222/// Simulate a user registering properties
223pub fn simulate_user_registration(
224    user_id: usize,
225    num_properties: usize,
226    config: &LoadTestConfig,
227    metrics: &LoadTestMetrics,
228) {
229    // Set caller for this user
230    let accounts = default_accounts::<DefaultEnvironment>();
231    let user_account = match user_id % 5 {
232        0 => accounts.alice,
233        1 => accounts.bob,
234        2 => accounts.charlie,
235        3 => accounts.django,
236        _ => accounts.eve,
237    };
238    set_caller::<DefaultEnvironment>(user_account);
239
240    let mut registry = PropertyRegistryContract::new();
241
242    for i in 0..num_properties {
243        let start = Instant::now();
244
245        let metadata = generate_property_metadata(user_id, i);
246        let result = registry.register_property(metadata);
247
248        let elapsed = start.elapsed().as_millis();
249
250        match result {
251            Ok(_) => metrics.record_success(elapsed as u128),
252            Err(_) => metrics.record_failure(),
253        }
254
255        // Respect operation delay
256        if config.operation_delay_ms > 0 {
257            thread::sleep(Duration::from_millis(config.operation_delay_ms));
258        }
259    }
260}

262/// Simulate a user querying properties
263pub fn simulate_user_queries(
264    user_id: usize,
265    num_queries: usize,
266    config: &LoadTestConfig,
267    metrics: &LoadTestMetrics,
268    registry: &PropertyRegistryContract,
269) {
270    let accounts = default_accounts::<DefaultEnvironment>();
271    let user_account = match user_id % 5 {
272        0 => accounts.alice,
273        1 => accounts.bob,
274        2 => accounts.charlie,
275        3 => accounts.django,
276        _ => accounts.eve,
277    };
278    set_caller::<DefaultEnvironment>(user_account);
279
280    for i in 0..num_queries {
281        let start = Instant::now();
282
283        // Query different property IDs
284        let property_id = i as u32;
285        let _result = registry.get_property(property_id as u64);
286
287        let elapsed = start.elapsed().as_millis();
288        metrics.record_success(elapsed as u128);
289
290        if config.operation_delay_ms > 0 {
291            thread::sleep(Duration::from_millis(config.operation_delay_ms));
292        }
293    }

296pub fn run_concurrent_load_test<F>(
297    config: &LoadTestConfig,
298    test_name: &str,
299    user_task: F,
300) -> LoadTestMetrics
301where
302    F: Fn(usize, &LoadTestConfig, &LoadTestMetrics) + Send + Sync + 'static,
303{
304    let metrics = LoadTestMetrics::default();
305    let start_time = Instant::now();
306
307    println!("\n🚀 Starting Load Test: {}", test_name);
308    println!("Configuration:");
309    println!("  Concurrent Users: {}", config.concurrent_users);
310    println!("  Duration: {} seconds", config.duration_secs);
311    println!("  Ramp-up: {} seconds", config.ramp_up_secs);
312    println!("  Target Ops/sec: {}", config.target_ops_per_second);
313
314    let mut handles = vec![];
315    let task_fn = Arc::new(user_task);
316
317    // Spawn concurrent user threads
318    for user_id in 0..config.concurrent_users {
319        let config_clone = config.clone();
320        let metrics_clone = LoadTestMetrics {
321            total_operations: Arc::clone(&metrics.total_operations),
322            successful_operations: Arc::clone(&metrics.successful_operations),
323            failed_operations: Arc::clone(&metrics.failed_operations),
324            total_response_time_ms: Arc::clone(&metrics.total_response_time_ms),
325            min_response_time_ms: Arc::clone(&metrics.min_response_time_ms),
326            max_response_time_ms: Arc::clone(&metrics.max_response_time_ms),
327            ops_per_second: Arc::clone(&metrics.ops_per_second),
328            peak_memory_mb: Arc::clone(&metrics.peak_memory_mb),
329        };
330        let task_fn_clone = Arc::clone(&task_fn);
331
332        let handle = thread::spawn(move || {
333            task_fn_clone(user_id, &config_clone, &metrics_clone);
334        });
335
336        handles.push(handle);
337
338        // Ramp-up delay
339        if config.ramp_up_secs > 0 {
340            let ramp_delay = Duration::from_millis(
341                (config.ramp_up_secs * 1000) / config.concurrent_users as u64,
342            );
343            thread::sleep(ramp_delay);
344        }
345    }
346
347    // Wait for all threads to complete
348    for handle in handles {
349        handle.join().expect("Thread should complete successfully");
350    }
351
352    // Calculate final metrics
353    let total_duration = start_time.elapsed().as_secs_f64();
354    let total_ops = *metrics.total_operations.lock().unwrap() as f64;
355    *metrics.ops_per_second.lock().unwrap() = total_ops / total_duration;
356
357    metrics.print_summary(test_name);
358
359    metrics
360}

362/// Assert that metrics meet performance thresholds
363pub fn assert_performance_thresholds(
364    metrics: &LoadTestMetrics,
365    test_name: &str,
366    max_avg_response_ms: f64,
367    min_success_rate: f64,
368    min_ops_per_second: f64,
369) {
370    let avg_response = metrics.avg_response_time_ms();
371    let success_rate = metrics.success_rate();
372    let ops_sec = *metrics.ops_per_second.lock().unwrap();
373
374    println!("\n📊 Performance Threshold Check: {}", test_name);
375    println!(
376        "  Avg Response: {:.2}ms (max: {:.2}ms)",
377        avg_response, max_avg_response_ms
378    );
379    println!(
380        "  Success Rate: {:.2}% (min: {:.2}%)",
381        success_rate, min_success_rate
382    );
383    println!(
384        "  Ops/Second: {:.2} (min: {:.2})",
385        ops_sec, min_ops_per_second
386    );
387
388    assert!(
389        avg_response <= max_avg_response_ms,
390        "Average response time {:.2}ms exceeds threshold {:.2}ms",
391        avg_response,
392        max_avg_response_ms
393    );
394
395    assert!(
396        success_rate >= min_success_rate,
397        "Success rate {:.2}% below threshold {:.2}%",
398        success_rate,
399        min_success_rate
400    );
401
402    assert!(
403        ops_sec >= min_ops_per_second,
404        "Operations/second {:.2} below threshold {:.2}",
405        ops_sec,
406        min_ops_per_second
407    );
408
409    println!("✅ All performance thresholds met!");
