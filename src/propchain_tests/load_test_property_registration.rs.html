12/// Test concurrent property registration with light load
13#[test]
14fn load_test_concurrent_registration_light() {
15    let config = LoadTestConfig::light();
16    
17    let metrics = run_concurrent_load_test(
18        &config,
19        "Concurrent Registration - Light Load",
20        |user_id, cfg, m| {
21            simulate_user_registration(user_id, 10, cfg, m);
22        },
23    );
24    
25    assert_performance_thresholds(
26        &metrics,
27        "Light Load Registration",
28        500.0,  // max avg response 500ms
29        95.0,   // min 95% success rate
30        20.0,   // min 20 ops/sec
31    );
32}

103/// Test registration with mixed read/write operations
104#[test]
105fn load_test_mixed_operations() {
106    let config = LoadTestConfig::medium();
107    
108    // First, register some properties
109    let accounts = default_accounts::<DefaultEnvironment>();
110    set_caller::<DefaultEnvironment>(accounts.alice);
111    let mut registry = PropertyRegistryContract::new();
112    
113    println!("📦 Pre-registering properties for mixed test...");
114    for i in 0..100 {
115        let metadata = generate_property_metadata(0, i);
116        registry.register_property(metadata).expect("Should register");
117    }
118    
119    let metrics = LoadTestMetrics::default();
120    let start_time = Instant::now();
121    
122    println!("🔄 Starting mixed operations load test...");
123    
124    // Simulate 70% reads, 30% writes
125    let num_writers = (config.concurrent_users * 30) / 100;
126    let num_readers = config.concurrent_users - num_writers;
127    
128    let mut handles = vec![];
129    
130    // Writer threads
131    for user_id in 0..num_writers {
132        let cfg = config.clone();
133        let m = LoadTestMetrics {
134            total_operations: Arc::clone(&metrics.total_operations),
135            successful_operations: Arc::clone(&metrics.successful_operations),
136            failed_operations: Arc::clone(&metrics.failed_operations),
137            total_response_time_ms: Arc::clone(&metrics.total_response_time_ms),
138            min_response_time_ms: Arc::clone(&metrics.min_response_time_ms),
139            max_response_time_ms: Arc::clone(&metrics.max_response_time_ms),
140            ops_per_second: Arc::clone(&metrics.ops_per_second),
141            peak_memory_mb: Arc::clone(&metrics.peak_memory_mb),
142        };
143        
144        let handle = thread::spawn(move || {
145            simulate_user_registration(user_id, 15, &cfg, &m);
146        });
147        handles.push(handle);
148    }
149    
150    // Reader threads
151    for user_id in num_writers..(num_writers + num_readers) {
152        let cfg = config.clone();
153        let m = LoadTestMetrics {
154            total_operations: Arc::clone(&metrics.total_operations),
155            successful_operations: Arc::clone(&metrics.successful_operations),
156            failed_operations: Arc::clone(&metrics.failed_operations),
157            total_response_time_ms: Arc::clone(&metrics.total_response_time_ms),
158            min_response_time_ms: Arc::clone(&metrics.min_response_time_ms),
159            max_response_time_ms: Arc::clone(&metrics.max_response_time_ms),
160            ops_per_second: Arc::clone(&metrics.ops_per_second),
161            peak_memory_mb: Arc::clone(&metrics.peak_memory_mb),
162        };
163        
164        let handle = thread::spawn(move || {
165            simulate_user_queries(user_id, 30, &cfg, &m, &registry);
166        });
167        handles.push(handle);
168    }
169    
170    // Wait for completion
171    for handle in handles {
172        handle.join().expect("Thread should complete");
173    }
174    
175    let total_duration = start_time.elapsed().as_secs_f64();
176    let total_ops = *metrics.total_operations.lock().unwrap() as f64;
177    *metrics.ops_per_second.lock().unwrap() = total_ops / total_duration;
178    
179    metrics.print_summary("Mixed Operations Load Test");
180    
181    assert_performance_thresholds(
182        &metrics,
183        "Mixed Read/Write Operations",
184        600.0,  // max avg response 600ms
185        93.0,   // min 93% success rate
186        80.0,   // min 80 ops/sec
187    );
188}
