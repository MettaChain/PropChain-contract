4const STATS: Symbol = symbol_short!("STATS");

6#[contracttype]
7#[derive(Clone, Default, Debug)]
8pub struct LoanStats {
9    pub total_loaned: i128,
10    pub active_loans: u32,
11    pub defaults: u32,
12}
13
14#[contract]
15pub struct LoanAnalyticsContract;
16
17#[contractimpl]
18impl LoanAnalyticsContract {
19    pub fn record_loan(env: Env, amount: i128) {
20        let mut stats: LoanStats = env.storage().instance().get(&STATS).unwrap_or_default();
21
22        stats.total_loaned += amount;
23        stats.active_loans += 1;
24
25        env.storage().instance().set(&STATS, &stats);
26    }
27
28    pub fn record_default(env: Env) {
29        let mut stats: LoanStats = env.storage().instance().get(&STATS).unwrap_or_default();
30
31        if stats.active_loans > 0 {
32            stats.active_loans -= 1;
33        }
34        stats.defaults += 1;
35
36        env.storage().instance().set(&STATS, &stats);
37    }
38
39    pub fn get_stats(env: Env) -> LoanStats {
40        env.storage().instance().get(&STATS).unwrap_or_default()
41    }
