225        pub fn create_campaign(&mut self, title: String, target_amount: u128) -> Result<u64, CrowdfundingError> {
226            self.campaign_count += 1;
227            let campaign = Campaign {
228                campaign_id: self.campaign_count,
229                creator: self.env().caller(),
230                title,
231                target_amount,
232                raised_amount: 0,
233                status: CampaignStatus::Draft,
234                investor_count: 0,
235            };
236            self.campaigns.insert(self.campaign_count, &campaign);
237            self.env().emit_event(CampaignCreated {

239                creator: self.env().caller(),

246        pub fn activate_campaign(&mut self, campaign_id: u64) -> Result<(), CrowdfundingError> {
247            let mut campaign = self.campaigns.get(campaign_id).ok_or(CrowdfundingError::CampaignNotFound)?;
248            if self.env().caller() != campaign.creator && self.env().caller() != self.admin {
249                return Err(CrowdfundingError::Unauthorized);

252            self.campaigns.insert(campaign_id, &campaign);
253            Ok(())

269        #[ink(message)]
270        pub fn invest(&mut self, campaign_id: u64, amount: u128) -> Result<(), CrowdfundingError> {
271            let caller = self.env().caller();
272            let profile = self.investor_profiles.get(caller).ok_or(CrowdfundingError::InvestorNotCompliant)?;
273            if profile.kyc_status != ComplianceStatus::Approved {
274                return Err(CrowdfundingError::InvestorNotCompliant);
275            }
276            if self.blocked_jurisdictions.contains(&profile.jurisdiction) {
277                return Err(CrowdfundingError::InvestorNotCompliant);
278            }
279            let mut campaign = self.campaigns.get(campaign_id).ok_or(CrowdfundingError::CampaignNotFound)?;
280            if campaign.status != CampaignStatus::Active {
281                return Err(CrowdfundingError::CampaignNotActive);
282            }
283            let current = self.investments.get((campaign_id, caller)).unwrap_or(0);
284            if current == 0 {
285                campaign.investor_count += 1;
286            }
287            self.investments.insert((campaign_id, caller), &(current + amount));
288            campaign.raised_amount += amount;
289            if campaign.raised_amount >= campaign.target_amount {
290                campaign.status = CampaignStatus::Funded;
291            }
292            self.campaigns.insert(campaign_id, &campaign);
293            let shares = (amount / 1000) as u64;
294            let current_shares = self.share_holdings.get((campaign_id, caller)).unwrap_or(0);
295            self.share_holdings.insert((campaign_id, caller), &(current_shares + shares));
296            self.env().emit_event(InvestmentMade {
297                campaign_id,
298                investor: caller,
299                amount,
300            });
301            Ok(())
302        }

348        #[ink(message)]
349        pub fn distribute_profit(&self, campaign_id: u64, total_profit: u128, investor: AccountId) -> u128 {
350            let campaign = self.campaigns.get(campaign_id).unwrap_or(Campaign {
351                campaign_id: 0,
352                creator: AccountId::from([0x0; 32]),
353                title: String::new(),
354                target_amount: 0,
355                raised_amount: 1,
356                status: CampaignStatus::Draft,
357                investor_count: 0,
358            });
359            let investment = self.investments.get((campaign_id, investor)).unwrap_or(0);
360            if campaign.target_amount == 0 {
361                return 0;
362            }
363            (total_profit * investment) / campaign.target_amount
364        }

386        #[ink(message)]
387        pub fn vote(&mut self, proposal_id: u64, in_favour: bool) -> Result<(), CrowdfundingError> {
388            let caller = self.env().caller();
389            if self.votes_cast.get((proposal_id, caller)).unwrap_or(false) {
390                return Err(CrowdfundingError::AlreadyVoted);
391            }
392            let mut proposal = self.proposals.get(proposal_id).ok_or(CrowdfundingError::ProposalNotFound)?;
393            if proposal.status != ProposalStatus::Active {
394                return Err(CrowdfundingError::ProposalNotActive);
395            }
396            let weight = self.voting_weights.get((proposal.campaign_id, caller)).unwrap_or(1);
397            if in_favour {
398                proposal.votes_for += weight;
399            } else {
400                proposal.votes_against += weight;
401            }
402            self.proposals.insert(proposal_id, &proposal);
403            self.votes_cast.insert((proposal_id, caller), &true);
404            Ok(())
405        }

407        #[ink(message)]
408        pub fn finalize_proposal(&mut self, proposal_id: u64) -> Result<ProposalStatus, CrowdfundingError> {
409            let mut proposal = self.proposals.get(proposal_id).ok_or(CrowdfundingError::ProposalNotFound)?;
410            proposal.status = if proposal.votes_for > proposal.votes_against {
411                ProposalStatus::Passed
412            } else {
413                ProposalStatus::Rejected
414            };
415            self.proposals.insert(proposal_id, &proposal);
416            Ok(proposal.status)
417        }

419        #[ink(message)]
420        pub fn list_shares(&mut self, campaign_id: u64, shares: u64, price_per_share: u128) -> Result<u64, CrowdfundingError> {
421            let caller = self.env().caller();
422            let held = self.share_holdings.get((campaign_id, caller)).unwrap_or(0);
423            if held < shares {
424                return Err(CrowdfundingError::InsufficientShares);
425            }
426            self.listing_count += 1;
427            let listing = ShareListing {
428                listing_id: self.listing_count,
429                seller: caller,
430                campaign_id,
431                shares,
432                price_per_share,
433            };
434            self.listings.insert(self.listing_count, &listing);
435            self.env().emit_event(SharesListed {
436                listing_id: self.listing_count,
437                seller: caller,
438                shares,
439            });
440            Ok(self.listing_count)
441        }

443        #[ink(message)]
444        pub fn buy_shares(&mut self, listing_id: u64) -> Result<u128, CrowdfundingError> {
445            let listing = self.listings.get(listing_id).ok_or(CrowdfundingError::ListingNotFound)?;
446            let total_cost = listing.price_per_share * listing.shares as u128;
447            let seller_shares = self.share_holdings.get((listing.campaign_id, listing.seller)).unwrap_or(0);
448            self.share_holdings.insert((listing.campaign_id, listing.seller), &seller_shares.saturating_sub(listing.shares));
449            let buyer = self.env().caller();
450            let buyer_shares = self.share_holdings.get((listing.campaign_id, buyer)).unwrap_or(0);
451            self.share_holdings.insert((listing.campaign_id, buyer), &(buyer_shares + listing.shares));
452            self.listings.remove(listing_id);
453            Ok(total_cost)
454        }

456        #[ink(message)]
457        pub fn assess_risk(&mut self, campaign_id: u64, ltv: u32, dev_score: u32, volatility: u32) -> Result<(), CrowdfundingError> {
458            if self.env().caller() != self.admin {
459                return Err(CrowdfundingError::Unauthorized);
460            }
461            let rating = if ltv < 60 && dev_score >= 75 && volatility < 15 {
462                RiskRating::Low
463            } else if ltv < 80 && dev_score >= 50 && volatility < 30 {
464                RiskRating::Medium
465            } else {
466                RiskRating::High
467            };
468            let profile = RiskProfile {
469                campaign_id,
470                ltv_ratio: ltv,
471                developer_score: dev_score,
472                market_volatility: volatility,
473                rating,
474            };
475            self.risk_profiles.insert(campaign_id, &profile);
476            Ok(())
477        }
