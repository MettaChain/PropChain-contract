186        /// Initiates a bridge request with multi-signature requirement
187        #[ink(message)]
188        pub fn initiate_bridge_multisig(
189            &mut self,
190            token_id: TokenId,
191            destination_chain: ChainId,
192            recipient: AccountId,
193            required_signatures: u8,
194            timeout_blocks: Option<u64>,
195            metadata: PropertyMetadata,
196        ) -> Result<u64, Error> {
197            let caller = self.env().caller();
198
199            // Check if bridge is paused
200            if self.config.emergency_pause {
201                return Err(Error::BridgePaused);
202            }
203
204            // Validate destination chain
205            if !self.config.supported_chains.contains(&destination_chain) {
206                return Err(Error::InvalidChain);
207            }
208
209            // Validate signature requirements
210            if required_signatures < self.config.min_signatures_required
211                || required_signatures > self.config.max_signatures_required
212            {
213                return Err(Error::InsufficientSignatures);
214            }
215
216            // Check if caller is authorized (token owner or approved operator)
217            if !self.is_authorized_for_token(caller, token_id) {
218                return Err(Error::Unauthorized);
219            }
220
221            // Enforce rate limiting
222            // For NFT bridge, we count requests but value is 0 here since NFT value isn't strictly defined by amount.
223            self.check_and_update_rate_limits(caller, destination_chain, 0, true)?;
224
225            // Create bridge request
226            self.request_counter += 1;
227            let request_id = self.request_counter;
228            let current_block = u64::from(self.env().block_number());
229            let expires_at = timeout_blocks.map(|blocks| current_block + blocks);
230
231            let request = MultisigBridgeRequest {
232                request_id,
233                token_id,
234                source_chain: self.get_current_chain_id(),
235                destination_chain,
236                sender: caller,
237                recipient,
238                required_signatures,
239                signatures: Vec::new(),
240                created_at: current_block,
241                expires_at,
242                status: BridgeOperationStatus::Pending,
243                metadata,
244            };
245
246            self.bridge_requests.insert(request_id, &request);
247
248            self.env().emit_event(BridgeRequestCreated {
249                request_id,
250                token_id,
251                source_chain: request.source_chain,
252                destination_chain,
253                requester: caller,
254            });
255
256            Ok(request_id)
257        }

259        /// Signs a bridge request
260        #[ink(message)]
261        pub fn sign_bridge_request(&mut self, request_id: u64, approve: bool) -> Result<(), Error> {
262            let caller = self.env().caller();
263
264            // Check if caller is a bridge operator
265            if !self.bridge_operators.contains(&caller) {
266                return Err(Error::Unauthorized);
267            }
268
269            let mut request = self
270                .bridge_requests
271                .get(request_id)
272                .ok_or(Error::InvalidRequest)?;
273
274            // Check if request has expired
275            if let Some(expires_at) = request.expires_at {
276                if u64::from(self.env().block_number()) > expires_at {
277                    return Err(Error::RequestExpired);
278                }
279            }
280
281            // Check if already signed
282            if request.signatures.contains(&caller) {
283                return Err(Error::AlreadySigned);
284            }
285
286            // Add signature
287            request.signatures.push(caller);
288
289            // Update status based on approval and signatures collected
290            if !approve {
291                request.status = BridgeOperationStatus::Failed;
292            } else if request.signatures.len() >= request.required_signatures as usize {
293                request.status = BridgeOperationStatus::Locked;
294            }
295
296            self.bridge_requests.insert(request_id, &request);
297
298            self.env().emit_event(BridgeRequestSigned {
299                request_id,
300                signer: caller,
301                signatures_collected: request.signatures.len() as u8,
302                signatures_required: request.required_signatures,
303            });
304
305            Ok(())
306        }

319        /// Sign a bridge request with optional ECDSA cryptographic signature verification.
320        #[ink(message)]
321        pub fn sign_bridge_request_with_signature(
322            &mut self,
323            request_id: u64,
324            approve: bool,
325            signed_approval: Option<propchain_traits::SignedApproval>,
326        ) -> Result<(), Error> {
327            let caller = self.env().caller();
328
329            if let Some(ref approval) = signed_approval {
330                let expected_key = self
331                    .operator_public_keys
332                    .get(caller)
333                    .ok_or(Error::Unauthorized)?;
334                propchain_traits::crypto::verify_signed_approval(approval, &expected_key)
335                    .map_err(|_| Error::Unauthorized)?;
336
337                let expected_hash = propchain_traits::crypto::hash_encoded(&(
338                    request_id,
339                    approve,
340                    caller,
341                    self.env().block_number(),
342                ));
343                if approval.message_hash != <[u8; 32]>::from(expected_hash) {
344                    return Err(Error::Unauthorized);
345                }
346            }
347
348            self.sign_bridge_request(request_id, approve)
349        }

351        /// Executes a bridge request after collecting required signatures
352        #[ink(message)]
353        pub fn execute_bridge(&mut self, request_id: u64) -> Result<(), Error> {
354            let caller = self.env().caller();
355
356            // Check if caller is a bridge operator
357            if !self.bridge_operators.contains(&caller) {
358                return Err(Error::Unauthorized);
359            }
360
361            let mut request = self
362                .bridge_requests
363                .get(request_id)
364                .ok_or(Error::InvalidRequest)?;
365
366            // Check if request is ready for execution
367            if request.status != BridgeOperationStatus::Locked {
368                return Err(Error::InvalidRequest);
369            }
370
371            // Check if enough signatures are collected
372            if request.signatures.len() < request.required_signatures as usize {
373                return Err(Error::InsufficientSignatures);
374            }
375
376            // Generate transaction hash
377            let transaction_hash = self.generate_transaction_hash(&request);
378
379            // Create bridge transaction record
380            self.transaction_counter += 1;
381            let transaction = BridgeTransaction {
382                transaction_id: self.transaction_counter,
383                token_id: request.token_id,
384                source_chain: request.source_chain,
385                destination_chain: request.destination_chain,
386                sender: request.sender,
387                recipient: request.recipient,
388                transaction_hash,
389                timestamp: self.env().block_timestamp(),
390                gas_used: self.estimate_gas_usage(&request),
391                status: BridgeOperationStatus::InTransit,
392                metadata: request.metadata.clone(),
393            };
394
395            // Update request status
396            request.status = BridgeOperationStatus::Completed;
397            self.bridge_requests.insert(request_id, &request);
398
399            // Store transaction verification
400            self.verified_transactions.insert(transaction_hash, &true);
401
402            // Add to bridge history
403            let mut history = self.bridge_history.get(request.sender).unwrap_or_default();
404            history.push(transaction.clone());
405            self.bridge_history.insert(request.sender, &history);
406
407            self.env().emit_event(BridgeExecuted {
408                request_id,
409                token_id: request.token_id,
410                transaction_hash,
411            });
412
413            Ok(())
414        }

416        /// Recovers from a failed bridge operation
417        #[ink(message)]
418        pub fn recover_failed_bridge(
419            &mut self,
420            request_id: u64,
421            recovery_action: RecoveryAction,
422        ) -> Result<(), Error> {
423            let caller = self.env().caller();
424
425            // Only admin can recover failed bridges
426            if caller != self.admin {
427                return Err(Error::Unauthorized);
428            }
429
430            let mut request = self
431                .bridge_requests
432                .get(request_id)
433                .ok_or(Error::InvalidRequest)?;
434
435            // Check if request is in a failed state
436            if !matches!(
437                request.status,
438                BridgeOperationStatus::Failed | BridgeOperationStatus::Expired
439            ) {
440                return Err(Error::InvalidRequest);
441            }
442
443            // Execute recovery action
444            match recovery_action {
445                RecoveryAction::UnlockToken => {
446                    // Logic to unlock the token would be implemented here
447                    // This would typically call back to the property token contract
448                }
449                RecoveryAction::RefundGas => {
450                    // Logic to refund gas costs would be implemented here
451                }
452                RecoveryAction::RetryBridge => {
453                    // Reset request to pending for retry
454                    request.status = BridgeOperationStatus::Pending;
455                    request.signatures.clear();
456                }
457                RecoveryAction::CancelBridge => {
458                    // Mark as cancelled
459                    request.status = BridgeOperationStatus::Failed;
460                }
461            }
462
463            self.bridge_requests.insert(request_id, &request);
464
465            self.env().emit_event(BridgeRecovered {
466                request_id,
467                recovery_action,
468            });
469
470            Ok(())
471        }

545        /// Registers a cross-chain DEX trade intent on the bridge.
546        #[ink(message)]
547        pub fn register_cross_chain_trade(
548            &mut self,
549            pair_id: u64,
550            order_id: Option<u64>,
551            destination_chain: ChainId,
552            recipient: AccountId,
553            amount_in: u128,
554            min_amount_out: u128,
555        ) -> Result<u64, Error> {
556            if self.config.emergency_pause {
557                return Err(Error::BridgePaused);
558            }
559            if !self.config.supported_chains.contains(&destination_chain) {
560                return Err(Error::InvalidChain);
561            }
562
563            // Enforce rate limiting
564            // For cross-chain trades, we track the volume (amount_in) but don't count it as an NFT request.
565            self.check_and_update_rate_limits(self.env().caller(), destination_chain, amount_in, false)?;
566
567            self.cross_chain_trade_counter += 1;
568            let trade_id = self.cross_chain_trade_counter;
569            let quote = self.quote_cross_chain_trade(destination_chain, amount_in)?;
570            let intent = CrossChainTradeIntent {
571                trade_id,
572                pair_id,
573                order_id,
574                source_chain: self.get_current_chain_id(),
575                destination_chain,
576                trader: self.env().caller(),
577                recipient,
578                amount_in,
579                min_amount_out,
580                bridge_request_id: None,
581                bridge_fee_quote: quote,
582                status: CrossChainTradeStatus::Pending,
583                created_at: self.env().block_timestamp(),
584            };
585            self.cross_chain_trades.insert(trade_id, &intent);
586            Ok(trade_id)
587        }

746        /// Confirm a pending admin rotation after cooldown.

748        pub fn confirm_admin_rotation(&mut self) -> Result<(), Error> {
749            let caller = self.env().caller();
750            let block = self.env().block_number();
751
752            let request = self
753                .pending_admin_rotation
754                .as_ref()
755                .ok_or(Error::InvalidRequest)?;
756
757            if request.new_account != caller {
758                return Err(Error::Unauthorized);
759            }
760            if block < request.effective_at {
761                return Err(Error::InvalidRequest);
762            }
763            let expiry = request
764                .effective_at
765                .saturating_add(propchain_traits::constants::KEY_ROTATION_EXPIRY_BLOCKS);
766            if block > expiry {
767                self.pending_admin_rotation = None;
768                return Err(Error::RequestExpired);
769            }
770
771            self.admin = caller;
772            self.pending_admin_rotation = None;
773            Ok(())
774        }

776        /// Cancel a pending admin rotation.
777        #[ink(message)]
778        pub fn cancel_admin_rotation(&mut self) -> Result<(), Error> {
779            let caller = self.env().caller();
780            let request = self
781                .pending_admin_rotation
782                .as_ref()
783                .ok_or(Error::InvalidRequest)?;
784
785            if caller != request.old_account && caller != request.new_account {
786                return Err(Error::Unauthorized);
787            }
788
789            self.pending_admin_rotation = None;
790            Ok(())
791        }

827        fn check_and_update_rate_limits(
828            &mut self,
829            account: AccountId,
830            destination_chain: ChainId,
831            amount: u128,
832            is_nft: bool,
833        ) -> Result<(), Error> {
834            if !self.config.rate_limit_enabled {
835                return Ok(());
836            }
837
838            let current_day = self.env().block_timestamp() / 86_400_000;
839
840            if is_nft {
841                let last_reset = self.account_last_reset_day.get(account).unwrap_or(0);
842                let mut daily_requests = self.account_daily_requests.get(account).unwrap_or(0);
843
844                if last_reset < current_day {
845                    daily_requests = 0;
846                    self.account_last_reset_day.insert(account, &current_day);
847                }
848
849                if daily_requests >= self.config.max_requests_per_day {
850                    return Err(Error::RateLimitExceeded);
851                }
852
853                self.account_daily_requests.insert(account, &(daily_requests + 1));
854            }
855
856            if amount > 0 {
857                let chain_info = self
858                    .chain_info
859                    .get(destination_chain)
860                    .ok_or(Error::InvalidChain)?;
861                let last_chain_reset = self.chain_last_reset_day.get(destination_chain).unwrap_or(0);
862                let mut chain_volume = self.chain_daily_volume.get(destination_chain).unwrap_or(0);
863
864                if last_chain_reset < current_day {
865                    chain_volume = 0;
866                    self.chain_last_reset_day.insert(destination_chain, &current_day);
867                }
868
869                if chain_volume.saturating_add(amount) > chain_info.chain_daily_limit {
870                    return Err(Error::RateLimitExceeded);
871                }
872
873                self.chain_daily_volume
874                    .insert(destination_chain, &(chain_volume + amount));
875            }
876
877            Ok(())
878        }
