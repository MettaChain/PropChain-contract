3/// Error types for the property token contract
4#[derive(Debug, PartialEq, Eq, scale::Encode, scale::Decode)]
5#[cfg_attr(feature = "std", derive(scale_info::TypeInfo))]
6pub enum Error {
7    // Standard ERC errors
8    /// Token does not exist
9    TokenNotFound,
10    /// Caller is not authorized
11    Unauthorized,
12    // Property-specific errors
13    /// Property does not exist
14    PropertyNotFound,
15    /// Metadata is invalid or malformed
16    InvalidMetadata,
17    /// Document does not exist
18    DocumentNotFound,
19    /// Compliance check failed
20    ComplianceFailed,
21    // Cross-chain bridge errors
22    /// Bridge functionality not supported
23    BridgeNotSupported,
24    /// Invalid chain ID
25    InvalidChain,
26    /// Token is locked in bridge
27    BridgeLocked,
28    /// Insufficient signatures for bridge operation
29    InsufficientSignatures,
30    /// Bridge request has expired
31    RequestExpired,
32    /// Invalid bridge request
33    InvalidRequest,
34    /// Bridge operations are paused
35    BridgePaused,
36    /// Gas limit exceeded
37    GasLimitExceeded,
38    /// Metadata is corrupted
39    MetadataCorruption,
40    /// Invalid bridge operator
41    InvalidBridgeOperator,
42    /// Duplicate bridge request
43    DuplicateBridgeRequest,
44    /// Bridge operation timed out
45    BridgeTimeout,
46    /// Already signed this request
47    AlreadySigned,
48    /// Insufficient balance
49    InsufficientBalance,
50    /// Invalid amount
51    InvalidAmount,
52    /// Proposal not found
53    ProposalNotFound,
54    /// Proposal is closed
55    ProposalClosed,
56    /// Ask not found
57    AskNotFound,
58    /// Input batch exceeds maximum allowed size
59    BatchSizeExceeded,
60}
61
62impl core::fmt::Display for Error {
63    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
64        match self {
65            Error::TokenNotFound => write!(f, "Token does not exist"),
66            Error::Unauthorized => write!(f, "Caller is not authorized"),
67            Error::PropertyNotFound => write!(f, "Property does not exist"),
68            Error::InvalidMetadata => write!(f, "Metadata is invalid or malformed"),
69            Error::DocumentNotFound => write!(f, "Document does not exist"),
70            Error::ComplianceFailed => write!(f, "Compliance check failed"),
71            Error::BridgeNotSupported => write!(f, "Bridge functionality not supported"),
72            Error::InvalidChain => write!(f, "Invalid chain ID"),
73            Error::BridgeLocked => write!(f, "Token is locked in bridge"),
74            Error::InsufficientSignatures => {
75                write!(f, "Insufficient signatures for bridge operation")
76            }
77            Error::RequestExpired => write!(f, "Bridge request has expired"),
78            Error::InvalidRequest => write!(f, "Invalid bridge request"),
79            Error::BridgePaused => write!(f, "Bridge operations are paused"),
80            Error::GasLimitExceeded => write!(f, "Gas limit exceeded"),
81            Error::MetadataCorruption => write!(f, "Metadata is corrupted"),
82            Error::InvalidBridgeOperator => write!(f, "Invalid bridge operator"),
83            Error::DuplicateBridgeRequest => write!(f, "Duplicate bridge request"),
84            Error::BridgeTimeout => write!(f, "Bridge operation timed out"),
85            Error::AlreadySigned => write!(f, "Already signed this request"),
86            Error::InsufficientBalance => write!(f, "Insufficient balance"),
87            Error::InvalidAmount => write!(f, "Invalid amount"),
88            Error::ProposalNotFound => write!(f, "Proposal not found"),
89            Error::ProposalClosed => write!(f, "Proposal is closed"),
90            Error::AskNotFound => write!(f, "Ask not found"),
91            Error::BatchSizeExceeded => write!(f, "Input batch exceeds maximum allowed size"),
92        }
93    }
94}
