224        /// Request a new valuation for a property
225        #[ink(message)]
226        pub fn request_property_valuation(&mut self, property_id: u64) -> Result<u64, OracleError> {
227            // Check if request already pending
228            if let Some(timestamp) = self.pending_requests.get(&property_id) {
229                let current_time = self.env().block_timestamp();
230                if current_time.saturating_sub(timestamp) < self.max_price_staleness {
231                    return Err(OracleError::RequestPending);
232                }
233            }
234
235            let request_id = self.request_id_counter;
236            self.request_id_counter += 1;
237            
238            self.pending_requests.insert(&property_id, &self.env().block_timestamp());
239            
240            Ok(request_id)
241        }

298            if let Some(current) = self.property_valuations.get(&property_id) {

304                    if volatility < 10 { // 10% volatility

501        pub fn aggregate_prices(&self, prices: &[PriceData]) -> Result<u128, OracleError> {
502            if prices.len() < self.min_sources_required as usize {
503                return Err(OracleError::InsufficientSources);
504            }
505
506            // Remove outliers
507            let filtered_prices = self.filter_outliers(prices);
508
509            if filtered_prices.is_empty() {
510                return Err(OracleError::InsufficientSources);
511            }
512
513            // Weighted average based on source weights
514            let mut total_weighted_price = 0u128;
515            let mut total_weight = 0u32;
516
517            for price_data in &filtered_prices {
518                let weight = self.get_source_weight(&price_data.source)?;
519                total_weighted_price += price_data.price * weight as u128;
520                total_weight += weight;
521            }
522
523            if total_weight == 0 {
524                return Err(OracleError::InvalidParameters);
525            }
526
527            Ok(total_weighted_price / total_weight as u128)
528        }

530        pub fn filter_outliers(&self, prices: &[PriceData]) -> Vec<PriceData> {
531            if prices.len() < 3 {
532                return prices.to_vec();
533            }
534
535            // Calculate mean
536            let sum: u128 = prices.iter().map(|p| p.price).sum();
537            let mean = sum / prices.len() as u128;
538
539            // Calculate standard deviation using fixed point arithmetic
540            let variance: u128 = prices
541                .iter()
542                .map(|p| {
543                    let diff = p.price.abs_diff(mean);
544                    diff * diff
545                })
546                .sum();
547
548            let variance_avg = variance / prices.len() as u128;
549            // Integer square root via Newton-Raphson.
550            // Starting from variance_avg is always an upper bound (sqrt(x) <= x for x >= 1),
551            // so the sequence decreases monotonically to floor(sqrt(variance_avg)).
552            let std_dev = if variance_avg == 0 {
553                0u128
554            } else {
555                let mut x = variance_avg;
556                loop {
557                    let y = (x + variance_avg / x) / 2;
558                    if y >= x {
559                        break x; // converged
560                    }
561                    x = y;
562                }
563            };
564
565            // Filter outliers (beyond threshold standard deviations)
566            prices
567                .iter()
568                .filter(|p| {
569                    let diff = p.price.abs_diff(mean);
570                    diff <= std_dev * self.outlier_threshold as u128
571                })
572                .cloned()
573                .collect()
574        }

583        pub fn calculate_confidence_score(&self, prices: &[PriceData]) -> Result<u32, OracleError> {
584            if prices.is_empty() {
585                return Ok(0);
586            }
587
588            // Simple confidence based on number of sources and price variance
589            let source_confidence = (prices.len() as u32 * 25).min(75); // Max 75 from sources
590
591            // Calculate coefficient of variation
592            let sum: u128 = prices.iter().map(|p| p.price).sum();
593            let mean = sum / prices.len() as u128;
594
595            let variance: u128 = prices
596                .iter()
597                .map(|p| {
598                    let diff = p.price.abs_diff(mean);
599                    diff * diff
600                })
601                .sum();
602
603            // Calculate coefficient of variation using fixed point arithmetic
604            let std_dev = if !prices.is_empty() {
605                let variance_avg = variance / prices.len() as u128;
606                // Simple approximation of square root (for fixed point)
607                let mut approx = variance_avg;
608                for _ in 0..5 {
609                    // Newton-Raphson approximation
610                    if approx > 0 {
611                        approx = (approx + variance_avg / approx) / 2;
612                    }
613                }
614                approx
615            } else {
616                0
617            };
618
619            let cv = if mean > 0 {
620                (std_dev * 10000) / mean // Multiply by 10000 for precision
621            } else {
622                10000
623            };
624
625            // Lower CV = higher confidence (CV is in basis points)
626            let variance_confidence = if cv <= 10000 {
627                ((10000 - cv) / 400) as u32 // Scale to 0-25 range
628            } else {
629                0
630            };
631
632            Ok(source_confidence + variance_confidence)
633        }

635        fn calculate_volatility(&self, property_id: u64) -> Result<u32, OracleError> {
636            let historical = self.get_historical_valuations(property_id, 30); // Last 30 valuations
637
638            if historical.len() < 2 {
639                return Ok(0);
640            }
641
642            // Calculate price changes
643            let mut changes = Vec::new();
644            for i in 1..historical.len() {
645                let prev = historical[i - 1].valuation;
646                let curr = historical[i].valuation;
647
648                if prev > 0 {
649                    let change = (curr.abs_diff(prev) * 10000) / prev;
650                    changes.push(change);
651                }
652            }
653
654            // Average absolute change as volatility index (in basis points)
655            let total_change: u128 = changes.iter().sum();
656            let avg_change_bp = total_change / changes.len() as u128;
657            Ok((avg_change_bp / 100).min(100) as u32) // Convert to percentage
658        }

715                            });
716                        }

779    /// Implementation of the OracleRegistry trait from propchain-traits
780    impl propchain_traits::OracleRegistry for PropertyValuationOracle {
781        #[ink(message)]
782        fn add_source(&mut self, source: OracleSource) -> Result<(), OracleError> {
783            self.add_oracle_source(source)
784        }
785
786        #[ink(message)]
787        fn remove_source(&mut self, source_id: String) -> Result<(), OracleError> {
788            self.ensure_admin()?;
789            self.oracle_sources.remove(&source_id);
790            self.active_sources.retain(|id| id != &source_id);
791            Ok(())
792        }
793
794        #[ink(message)]
795        fn update_reputation(&mut self, source_id: String, success: bool) -> Result<(), OracleError> {
796            self.update_source_reputation(source_id, success)
797        }
798
799        #[ink(message)]
800        fn get_reputation(&self, source_id: String) -> Option<u32> {
801            self.source_reputations.get(&source_id)
802        }
803
804        #[ink(message)]
805        fn slash_source(&mut self, source_id: String, penalty_amount: u128) -> Result<(), OracleError> {
806            self.slash_source(source_id, penalty_amount)
807        }
808
809        #[ink(message)]
810        fn detect_anomalies(&self, property_id: u64, new_valuation: u128) -> bool {
811            self.is_anomaly(property_id, new_valuation)
812        }
813    }
