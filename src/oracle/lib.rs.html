163            self.ensure_admin()?;
164

179            // Emit event
180            self.env().emit_event(ValuationUpdated {
181                property_id,
182                valuation: valuation.valuation,
183                confidence_score: valuation.confidence_score,
184                timestamp: self.env().block_timestamp(),

192        pub fn update_valuation_from_sources(
193            &mut self,

196            // Collect prices from all active sources
197            let prices = self.collect_prices_from_sources(property_id)?;

207            let valuation = PropertyValuation {
208                property_id,
209                valuation: aggregated_price,
210                confidence_score,
211                sources_used: prices.len() as u32,
212                last_updated: self.env().block_timestamp(),
213                valuation_method: ValuationMethod::MarketData,
214            };
215

217        }

404        pub fn aggregate_prices(&self, prices: &[PriceData]) -> Result<u128, OracleError> {
405            if prices.len() < self.min_sources_required as usize {
406                return Err(OracleError::InsufficientSources);
407            }
408
409            // Remove outliers
410            let filtered_prices = self.filter_outliers(prices);
411
412            if filtered_prices.is_empty() {
413                return Err(OracleError::InsufficientSources);
414            }
415
416            // Weighted average based on source weights
417            let mut total_weighted_price = 0u128;
418            let mut total_weight = 0u32;
419
420            for price_data in &filtered_prices {
421                let weight = self.get_source_weight(&price_data.source)?;
422                total_weighted_price += price_data.price * weight as u128;
423                total_weight += weight;
424            }
425
426            if total_weight == 0 {
427                return Err(OracleError::InvalidParameters);
428            }
429
430            Ok(total_weighted_price / total_weight as u128)
431        }

433        pub fn filter_outliers(&self, prices: &[PriceData]) -> Vec<PriceData> {
434            if prices.len() < 3 {
435                return prices.to_vec();
436            }
437
438            // Calculate mean
439            let sum: u128 = prices.iter().map(|p| p.price).sum();
440            let mean = sum / prices.len() as u128;
441
442            // Calculate standard deviation using fixed point arithmetic
443            let variance: u128 = prices
444                .iter()
445                .map(|p| {
446                    let diff = p.price.abs_diff(mean);
447                    diff * diff
448                })
449                .sum();
450
451            let variance_avg = variance / prices.len() as u128;
452            // Integer square root via Newton-Raphson.
453            // Starting from variance_avg is always an upper bound (sqrt(x) <= x for x >= 1),
454            // so the sequence decreases monotonically to floor(sqrt(variance_avg)).
455            let std_dev = if variance_avg == 0 {
456                0u128
457            } else {
458                let mut x = variance_avg;
459                loop {
460                    let y = (x + variance_avg / x) / 2;
461                    if y >= x {
462                        break x; // converged
463                    }
464                    x = y;
465                }
466            };
467
468            // Filter outliers (beyond threshold standard deviations)
469            prices
470                .iter()
471                .filter(|p| {
472                    let diff = p.price.abs_diff(mean);
473                    diff <= std_dev * self.outlier_threshold as u128
474                })
475                .cloned()
476                .collect()
477        }

486        pub fn calculate_confidence_score(&self, prices: &[PriceData]) -> Result<u32, OracleError> {
487            if prices.is_empty() {
488                return Ok(0);
489            }
490
491            // Simple confidence based on number of sources and price variance
492            let source_confidence = (prices.len() as u32 * 25).min(75); // Max 75 from sources
493
494            // Calculate coefficient of variation
495            let sum: u128 = prices.iter().map(|p| p.price).sum();
496            let mean = sum / prices.len() as u128;
497
498            let variance: u128 = prices
499                .iter()
500                .map(|p| {
501                    let diff = p.price.abs_diff(mean);
502                    diff * diff
503                })
504                .sum();
505
506            // Calculate coefficient of variation using fixed point arithmetic
507            let std_dev = if !prices.is_empty() {
508                let variance_avg = variance / prices.len() as u128;
509                // Simple approximation of square root (for fixed point)
510                let mut approx = variance_avg;
511                for _ in 0..5 {
512                    // Newton-Raphson approximation
513                    if approx > 0 {
514                        approx = (approx + variance_avg / approx) / 2;
515                    }
516                }
517                approx
518            } else {
519                0
520            };
521
522            let cv = if mean > 0 {
523                (std_dev * 10000) / mean // Multiply by 10000 for precision
524            } else {
525                10000
526            };
527
528            // Lower CV = higher confidence (CV is in basis points)
529            let variance_confidence = if cv <= 10000 {
530                ((10000 - cv) / 400) as u32 // Scale to 0-25 range
531            } else {
532                0
533            };
534
535            Ok(source_confidence + variance_confidence)
536        }

538        fn calculate_volatility(&self, property_id: u64) -> Result<u32, OracleError> {
539            let historical = self.get_historical_valuations(property_id, 30); // Last 30 valuations
540
541            if historical.len() < 2 {
542                return Ok(0);
543            }
544
545            // Calculate price changes
546            let mut changes = Vec::new();
547            for i in 1..historical.len() {
548                let prev = historical[i - 1].valuation;
549                let curr = historical[i].valuation;
550
551                if prev > 0 {
552                    let change = (curr.abs_diff(prev) * 10000) / prev;
553                    changes.push(change);
554                }
555            }
556
557            // Average absolute change as volatility index (in basis points)
558            let total_change: u128 = changes.iter().sum();
559            let avg_change_bp = total_change / changes.len() as u128;
560            Ok((avg_change_bp / 100).min(100) as u32) // Convert to percentage
561        }

570            Ok((

598        fn check_price_alerts(
599            &mut self,
600            property_id: u64,
601            new_valuation: u128,
602        ) -> Result<(), OracleError> {
603            if let Some(last_valuation) = self.property_valuations.get(&property_id) {
604                let change_percentage =
605                    self.calculate_percentage_change(last_valuation.valuation, new_valuation);
606
607                if let Some(alerts) = self.price_alerts.get(&property_id) {
608                    for alert in alerts {
609                        if alert.is_active
610                            && change_percentage >= alert.threshold_percentage as u128
611                        {
612                            self.env().emit_event(PriceAlertTriggered {
613                                property_id,
614                                old_valuation: last_valuation.valuation,
615                                new_valuation,
616                                change_percentage: change_percentage as u32,
617                                alert_address: alert.alert_address,
618                            });
619                        }
620                    }
621                }
622            }
623            Ok(())
624        }

626        pub fn calculate_percentage_change(&self, old_value: u128, new_value: u128) -> u128 {
627            if old_value == 0 {
628                return 0;
629            }
630
631            let diff = new_value.abs_diff(old_value);
632
633            (diff * 100) / old_value
634        }

816
817        // 5 tightly-clustered values + 1 extreme outlier.

819        // 1000's deviation (750) > 2 * 335 (670), so it is filtered.
820        // The 5 normal values are all within 2 and are kept.
821        let prices = vec![

824                timestamp: ink::env::block_timestamp::<DefaultEnvironment>(),

830                source: "source2".to_string(),
831            },
832            PriceData {
833                price: 100,
834                timestamp: ink::env::block_timestamp::<DefaultEnvironment>(),
835                source: "source3".to_string(),
836            },
837            PriceData {
838                price: 101,
839                timestamp: ink::env::block_timestamp::<DefaultEnvironment>(),
840                source: "source4".to_string(),
841            },
842            PriceData {
843                price: 102,
844                timestamp: ink::env::block_timestamp::<DefaultEnvironment>(),
845                source: "source5".to_string(),
