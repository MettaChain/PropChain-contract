3/// Simple (unweighted) median of a sample.
4pub fn simple_median(values: &mut Vec<u128>) -> u128 {
5    values.sort_unstable();
6    let n = values.len();
7    if n == 0 {
8        return 0;
9    }
10    if n % 2 == 1 {
11        values[n / 2]
12    } else {
13        (values[n / 2 - 1]).saturating_add(values[n / 2]) / 2
14    }
15}

17/// Weighted median: the value at the cumulative-weight midpoint.
18pub fn weighted_median(values: &[(u128, u32)]) -> u128 {
19    if values.is_empty() {
20        return 0;
21    }
22    let mut weighted_values = values.to_vec();
23    weighted_values.sort_by_key(|(v, _)| *v);
24    let total_weight: u32 = weighted_values.iter().map(|(_, w)| w).sum();
25    if total_weight == 0 {
26        return weighted_values.first().map_or(0, |(v, _)| *v);
27    }
28    let mut cumulative_weight: u32 = 0;
29    for (value, weight) in &weighted_values {
30        cumulative_weight = cumulative_weight.saturating_add(*weight);
31        if cumulative_weight >= total_weight / 2 {
32            return *value;
33        }
34    }
35    weighted_values.last().map_or(0, |(v, _)| *v)
36}

38/// Trimmed mean: drop `trim_percent%` from each end, average the rest.
39pub fn trimmed_mean(values: &mut Vec<u128>, trim_percent: u32) -> u128 {
40    values.sort_unstable();
41    let n = values.len();
42    if n == 0 {
43        return 0;
44    }
45    let trim_count = ((n as u32) * trim_percent / 100) as usize;
46    if trim_count * 2 >= n {
47        return values[n / 2];
48    }
49    let trimmed_values = &values[trim_count..n - trim_count];
50    if trimmed_values.is_empty() {
51        return 0;
52    }
53    let sum: u128 = trimmed_values.iter().sum();
54    sum / (trimmed_values.len() as u128)
55}
