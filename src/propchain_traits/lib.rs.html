53/// Advanced escrow trait with multi-signature and document custody
54pub trait AdvancedEscrow {
55    /// Error type for escrow operations
56    type Error;
57
58    /// Create an advanced escrow with multi-signature support
59    #[allow(clippy::too_many_arguments)]
60    fn create_escrow_advanced(
61        &mut self,
62        property_id: u64,
63        amount: u128,
64        buyer: AccountId,
65        seller: AccountId,
66        participants: Vec<AccountId>,
67        required_signatures: u8,
68        release_time_lock: Option<u64>,
69    ) -> Result<u64, Self::Error>;
70
71    /// Deposit funds to escrow
72    fn deposit_funds(&mut self, escrow_id: u64) -> Result<(), Self::Error>;
73
74    /// Release funds with multi-signature approval
75    fn release_funds(&mut self, escrow_id: u64) -> Result<(), Self::Error>;
76
77    /// Refund funds with multi-signature approval
78    fn refund_funds(&mut self, escrow_id: u64) -> Result<(), Self::Error>;
79
80    /// Upload document hash to escrow
81    fn upload_document(
82        &mut self,
83        escrow_id: u64,
84        document_hash: ink::primitives::Hash,
85        document_type: String,
86    ) -> Result<(), Self::Error>;
87
88    /// Verify a document
89    fn verify_document(
90        &mut self,
91        escrow_id: u64,
92        document_hash: ink::primitives::Hash,
93    ) -> Result<(), Self::Error>;
94
95    /// Add a condition to the escrow
96    fn add_condition(&mut self, escrow_id: u64, description: String) -> Result<u64, Self::Error>;
97
98    /// Mark a condition as met
99    fn mark_condition_met(&mut self, escrow_id: u64, condition_id: u64) -> Result<(), Self::Error>;
100
101    /// Sign approval for release or refund
102    fn sign_approval(
103        &mut self,
104        escrow_id: u64,
105        approval_type: ApprovalType,
106    ) -> Result<(), Self::Error>;
107
108    /// Raise a dispute
109    fn raise_dispute(&mut self, escrow_id: u64, reason: String) -> Result<(), Self::Error>;
110
111    /// Resolve a dispute (admin only)
112    fn resolve_dispute(&mut self, escrow_id: u64, resolution: String) -> Result<(), Self::Error>;
113
114    /// Emergency override (admin only)
115    fn emergency_override(
116        &mut self,
117        escrow_id: u64,
118        release_to_seller: bool,
119    ) -> Result<(), Self::Error>;
120}

238
239    /// Get gas estimation for bridge operation

241        &self,
242        token_id: TokenId,
243        destination_chain: ChainId,
244    ) -> Result<u64, Self::Error>;
245
246    /// Get bridge history for an account
247    fn get_bridge_history(&self, account: ink::primitives::AccountId) -> Vec<BridgeTransaction>;

399    /// Informational events: resource creation, normal state transitions
400    Info,
401    /// Warning events: unusual conditions that may need attention
402    Warning,
403    /// Error events: operation failures, rejected transactions
404    Error,
405    /// Critical events: security-related, admin changes, emergency actions
406    Critical,
407}
