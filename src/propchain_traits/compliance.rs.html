10/// Jurisdiction identifier for regulatory rules
11#[derive(Debug, Clone, Copy, PartialEq, Eq, scale::Encode, scale::Decode)]
12#[cfg_attr(
13    feature = "std",
14    derive(scale_info::TypeInfo, ink::storage::traits::StorageLayout)
15)]
16pub struct Jurisdiction {
17    pub code: u32,
18    pub country_code: [u8; 2],
19    pub region_code: u16,
20    pub locality_code: u16,
21}
22
23/// Transaction type for compliance rules engine
24#[derive(Debug, Clone, Copy, PartialEq, Eq, scale::Encode, scale::Decode)]
25#[cfg_attr(
26    feature = "std",
27    derive(scale_info::TypeInfo, ink::storage::traits::StorageLayout)
28)]
29pub enum ComplianceOperation {
30    RegisterProperty,
31    TransferProperty,
32    UpdateMetadata,
33    CreateEscrow,
34    ReleaseEscrow,
35    ListForSale,
36    Purchase,
37    BridgeTransfer,
38}
39
40/// Trait for compliance registry (used by PropertyRegistry for automated checks)
41#[ink::trait_definition]
42pub trait ComplianceChecker {
43    /// Returns true if the account meets current compliance requirements
44    #[ink(message)]
45    fn is_compliant(&self, account: ink::primitives::AccountId) -> bool;
46}
47
48/// Trait for automated tax withholding in property transactions
49#[ink::trait_definition]
50pub trait TaxWithholder {
51    /// Calculate and withhold tax for a property transaction.
52    /// Returns the (withheld_amount, tax_collector).
53    #[ink(message)]
54    fn withhold_tax(
55        &mut self,
56        property_id: u64,
57        jurisdiction: Jurisdiction,
58        transaction_amount: u128,
59    ) -> (u128, ink::primitives::AccountId);
60}

66/// Log severity levels for classifying contract events.
67/// Used by off-chain tooling to filter and prioritize event streams.
68#[derive(Debug, Clone, Copy, PartialEq, Eq, scale::Encode, scale::Decode)]
69#[cfg_attr(
70    feature = "std",
71    derive(scale_info::TypeInfo, ink::storage::traits::StorageLayout)
72)]
73pub enum LogLevel {
74    /// Informational events: resource creation, normal state transitions
75    Info,
76    /// Warning events: unusual conditions that may need attention
77    Warning,
78    /// Error events: operation failures, rejected transactions
79    Error,
80    /// Critical events: security-related, admin changes, emergency actions
81    Critical,
82}
83
84/// Event categories for structured log aggregation and filtering.
85#[derive(Debug, Clone, Copy, PartialEq, Eq, scale::Encode, scale::Decode)]
86#[cfg_attr(
87    feature = "std",
88    derive(scale_info::TypeInfo, ink::storage::traits::StorageLayout)
89)]
90pub enum EventCategory {
91    /// Resource creation: property registered, escrow created, token minted
92    Lifecycle,
93    /// State mutations: transfers, metadata updates, status changes
94    StateChange,
95    /// Permission changes: approvals granted or revoked
96    Authorization,
97    /// Value movements: escrow releases, refunds, fee payments
98    Financial,
99    /// System operations: pause, resume, upgrades, config changes
100    Administrative,
101    /// Regulatory and compliance: verification, audit logs, consent
102    Audit,
103}
