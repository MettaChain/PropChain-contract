1//! Shared error handling framework for PropChain contracts
2//!
3//! This module provides a unified error handling system with:
4//! - Base error trait that all contract errors implement
5//! - Common error variants reusable across contracts
6//! - Numeric error codes for external API integration
7//! - Full Debug, Display, and From trait implementations
8//! - [`ErrorMessage`]: structured error snapshot combining code, category, message, and i18n key
9//! - [`ContractError::to_error_message()`]: default method to produce an `ErrorMessage`
10//! - [`ContractError::error_i18n_key()`]: default method returning a localization key
11
12use core::fmt;
13use scale::{Decode, Encode};
14
15#[cfg(feature = "std")]
16use scale_info::TypeInfo;

18// =============================================================================
19// Standardized Error Message
20// =============================================================================
21
22/// Structured snapshot of all error information for a single error instance.
23///
24/// Suitable for logging and client-side display. All string fields are `&'static str`
25/// for `no_std` / no-heap compatibility. This type is not SCALE-encoded since
26/// `&'static str` does not implement `Decode`; use it purely in-memory.
27#[derive(Debug, Clone, Copy, PartialEq, Eq)]
28pub struct ErrorMessage {
29    /// Numeric error code, globally unique across all PropChain contracts.
30    pub code: u32,
31    /// Top-level domain that produced this error.
32    pub category: ErrorCategory,
33    /// Short human-readable message (matches `error_description`).
34    pub message: &'static str,
35    /// Longer technical description suitable for logs and developer tooling.
36    pub description: &'static str,
37    /// Dot-separated localization key for client-side message lookup.
38    /// Format: `"<category>.<variant_snake_case>"`, e.g. `"compliance.not_verified"`.
39    pub i18n_key: &'static str,
40}
41
42// =============================================================================
43// Base Error Trait
44// =============================================================================
45
46/// Base trait for all PropChain contract errors.
47/// All contract-specific error enums must implement this trait.
48pub trait ContractError: fmt::Debug + fmt::Display + Encode + Decode {
49    /// Returns the numeric error code for this error variant.
50    /// Used for external API integration and monitoring.
51    fn error_code(&self) -> u32;
52
53    /// Returns a human-readable description of the error.
54    fn error_description(&self) -> &'static str;
55
56    /// Returns the category of this error.
57    fn error_category(&self) -> ErrorCategory {
58        match self.error_code() {
59            1..=999 => ErrorCategory::Common,
60            1000..=1999 => ErrorCategory::PropertyToken,
61            2000..=2999 => ErrorCategory::Escrow,
62            3000..=3999 => ErrorCategory::Bridge,
63            4000..=4999 => ErrorCategory::Oracle,
64            5000..=5999 => ErrorCategory::Fees,
65            6000..=6999 => ErrorCategory::Compliance,
66            7000..=7999 => ErrorCategory::Governance,
67            8000..=8999 => ErrorCategory::Staking,
68            9000..=9999 => ErrorCategory::Monitoring,
69            _ => ErrorCategory::Unknown,
70        }
71    }
72
73    /// Returns a dot-separated localization key for client-side message lookup.
74    ///
75    /// Format: `"<category>.<variant_snake_case>"`, e.g. `"compliance.not_verified"`.
76    /// Override this in each error type to provide a precise key.
77    fn error_i18n_key(&self) -> &'static str {
78        "unknown.error"
79    }
80
81    /// Constructs a complete [`ErrorMessage`] snapshot from this error.
82    /// No allocation is performed; all fields are `'static`.
83    fn to_error_message(&self) -> ErrorMessage {
84        ErrorMessage {
85            code: self.error_code(),
86            category: self.error_category(),
87            message: self.error_description(),
88            description: self.error_description(),
89            i18n_key: self.error_i18n_key(),
90        }
91    }
92}

133/// Common error variants that can be used across multiple contracts
134#[derive(Debug, Clone, Copy, PartialEq, Eq, Encode, Decode)]
135#[cfg_attr(feature = "std", derive(TypeInfo))]
136pub enum CommonError {
137    /// Unauthorized access - caller lacks required permissions
138    Unauthorized = 1,
139    /// Invalid parameters provided to function
140    InvalidParameters = 2,
141    /// Resource not found (generic)
142    NotFound = 3,
143    /// Insufficient funds or balance
144    InsufficientFunds = 4,
145    /// Operation not allowed in current state
146    InvalidState = 5,
147    /// Internal contract error
148    InternalError = 6,
149    /// Serialization/deserialization error
150    CodecError = 7,
151    /// Feature not yet implemented
152    NotImplemented = 8,
153    /// Operation timed out
154    Timeout = 9,
155    /// Duplicate operation or resource
156    Duplicate = 10,
157}
158
159impl fmt::Display for CommonError {
160    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
161        match self {
162            CommonError::Unauthorized => {
163                write!(f, "Unauthorized: caller lacks required permissions")
164            }
165            CommonError::InvalidParameters => write!(f, "Invalid parameters provided to function"),
166            CommonError::NotFound => write!(f, "Resource not found"),
167            CommonError::InsufficientFunds => write!(f, "Insufficient funds or balance"),
168            CommonError::InvalidState => write!(f, "Operation not allowed in current state"),
169            CommonError::InternalError => write!(f, "Internal contract error occurred"),
170            CommonError::CodecError => write!(f, "Serialization/deserialization error"),
171            CommonError::NotImplemented => write!(f, "Feature not yet implemented"),
172            CommonError::Timeout => write!(f, "Operation timed out"),
173            CommonError::Duplicate => write!(f, "Duplicate operation or resource"),
174        }
175    }
176}
177
178impl ContractError for CommonError {
179    fn error_code(&self) -> u32 {
180        *self as u32
181    }
182
183    fn error_description(&self) -> &'static str {
184        match self {
185            CommonError::Unauthorized => {
186                "Caller does not have permission to perform this operation"
187            }
188            CommonError::InvalidParameters => "One or more function parameters are invalid",
189            CommonError::NotFound => "The requested resource does not exist",
190            CommonError::InsufficientFunds => "Account has insufficient balance for this operation",
191            CommonError::InvalidState => "Cannot perform this operation in the current state",
192            CommonError::InternalError => "An internal error occurred in the contract",
193            CommonError::CodecError => "Failed to encode or decode data",
194            CommonError::NotImplemented => "This feature is not yet implemented",
195            CommonError::Timeout => "The operation exceeded its time limit",
196            CommonError::Duplicate => "This operation or resource already exists",
197        }
198    }
199
200    fn error_category(&self) -> ErrorCategory {
201        ErrorCategory::Common
202    }
203
204    fn error_i18n_key(&self) -> &'static str {
205        match self {
206            CommonError::Unauthorized => "common.unauthorized",
207            CommonError::InvalidParameters => "common.invalid_parameters",
208            CommonError::NotFound => "common.not_found",
209            CommonError::InsufficientFunds => "common.insufficient_funds",
210            CommonError::InvalidState => "common.invalid_state",
211            CommonError::InternalError => "common.internal_error",
212            CommonError::CodecError => "common.codec_error",
213            CommonError::NotImplemented => "common.not_implemented",
214            CommonError::Timeout => "common.timeout",
215            CommonError::Duplicate => "common.duplicate",
216        }
217    }
218}

385#[cfg(test)]
386mod tests {
387    use super::*;
388
389    #[test]
390    fn common_error_i18n_keys_are_correct() {
391        assert_eq!(
392            CommonError::Unauthorized.error_i18n_key(),
393            "common.unauthorized"
394        );
395        assert_eq!(CommonError::NotFound.error_i18n_key(), "common.not_found");
396        assert_eq!(CommonError::Duplicate.error_i18n_key(), "common.duplicate");
397    }
398
399    #[test]
400    fn to_error_message_populates_all_fields() {
401        let msg = CommonError::Unauthorized.to_error_message();
402        assert_eq!(msg.code, common_codes::UNAUTHORIZED);
403        assert_eq!(msg.category, ErrorCategory::Common);
404        assert_eq!(msg.i18n_key, "common.unauthorized");
405        assert!(!msg.description.is_empty());
406    }
407
408    #[test]
409    fn oracle_batch_size_exceeded_constant_matches_value() {
410        assert_eq!(oracle_codes::ORACLE_BATCH_SIZE_EXCEEDED, 4012);
411    }
412
413    #[test]
414    fn compliance_codes_are_unique() {
415        let mut codes = vec![
416            compliance_codes::COMPLIANCE_UNAUTHORIZED,
417            compliance_codes::COMPLIANCE_NOT_VERIFIED,
418            compliance_codes::COMPLIANCE_CHECK_FAILED,
419            compliance_codes::COMPLIANCE_DOCUMENT_MISSING,
420            compliance_codes::COMPLIANCE_EXPIRED,
421            compliance_codes::COMPLIANCE_HIGH_RISK,
422            compliance_codes::COMPLIANCE_PROHIBITED_JURISDICTION,
423            compliance_codes::COMPLIANCE_ALREADY_VERIFIED,
424            compliance_codes::COMPLIANCE_CONSENT_NOT_GIVEN,
425            compliance_codes::COMPLIANCE_INVALID_RISK_SCORE,
426            compliance_codes::COMPLIANCE_JURISDICTION_NOT_SUPPORTED,
427            compliance_codes::COMPLIANCE_INVALID_DOCUMENT_TYPE,
428            compliance_codes::COMPLIANCE_DATA_RETENTION_EXPIRED,
429        ];
430        let len = codes.len();
431        codes.sort();
432        codes.dedup();
