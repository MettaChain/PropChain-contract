169/// Cross-chain bridge trait for property tokens
170pub trait PropertyTokenBridge {
171    /// Error type for bridge operations
172    type Error;
173
174    /// Lock a token for bridging to another chain
175    fn lock_token_for_bridge(
176        &mut self,
177        token_id: TokenId,
178        destination_chain: ChainId,
179        recipient: AccountId,
180    ) -> Result<(), Self::Error>;
181
182    /// Mint a bridged token from another chain
183    fn mint_bridged_token(
184        &mut self,
185        source_chain: ChainId,
186        original_token_id: TokenId,
187        recipient: AccountId,
188        metadata: PropertyMetadata,
189    ) -> Result<TokenId, Self::Error>;
190
191    /// Burn a bridged token when returning to original chain
192    fn burn_bridged_token(
193        &mut self,
194        token_id: TokenId,
195        destination_chain: ChainId,
196        recipient: AccountId,
197    ) -> Result<(), Self::Error>;
198
199    /// Unlock a token that was previously locked
200    fn unlock_token(&mut self, token_id: TokenId, recipient: AccountId) -> Result<(), Self::Error>;
201
202    /// Get bridge status for a token
203    fn get_bridge_status(&self, token_id: TokenId) -> Option<BridgeStatus>;
204
205    /// Verify bridge transaction hash
206    fn verify_bridge_transaction(
207        &self,
208        token_id: TokenId,
209        transaction_hash: ink::primitives::Hash,
210        source_chain: ChainId,
211    ) -> bool;
212
213    /// Add a bridge operator
214    fn add_bridge_operator(&mut self, operator: AccountId) -> Result<(), Self::Error>;
215
216    /// Remove a bridge operator
217    fn remove_bridge_operator(&mut self, operator: AccountId) -> Result<(), Self::Error>;
218
219    /// Check if an account is a bridge operator
220    fn is_bridge_operator(&self, account: AccountId) -> bool;
221
222    /// Get all bridge operators
223    fn get_bridge_operators(&self) -> Vec<AccountId>;
224}

226/// Advanced bridge trait with multi-signature and monitoring
227pub trait AdvancedBridge {
228    /// Error type for advanced bridge operations
229    type Error;
230
231    /// Initiate bridge with multi-signature requirement
232    fn initiate_bridge_multisig(
233        &mut self,
234        token_id: TokenId,
235        destination_chain: ChainId,
236        recipient: AccountId,
237        required_signatures: u8,
238        timeout_blocks: Option<u64>,
239    ) -> Result<u64, Self::Error>; // Returns bridge request ID
240
241    /// Sign a bridge request
242    fn sign_bridge_request(
243        &mut self,
244        bridge_request_id: u64,
245        approve: bool,
246    ) -> Result<(), Self::Error>;
247
248    /// Execute bridge after collecting required signatures
249    fn execute_bridge(&mut self, bridge_request_id: u64) -> Result<(), Self::Error>;
250
251    /// Monitor bridge status and handle errors
252    fn monitor_bridge_status(&self, bridge_request_id: u64) -> Option<BridgeMonitoringInfo>;
253
254    /// Recover from failed bridge operation
255    fn recover_failed_bridge(
256        &mut self,
257        bridge_request_id: u64,
258        recovery_action: RecoveryAction,
259    ) -> Result<(), Self::Error>;
260
261    /// Get gas estimation for bridge operation
262    fn estimate_bridge_gas(
263        &self,
264        token_id: TokenId,
265        destination_chain: ChainId,
266    ) -> Result<u64, Self::Error>;
267
268    /// Get bridge history for an account
269    fn get_bridge_history(&self, account: AccountId) -> Vec<BridgeTransaction>;
270}
