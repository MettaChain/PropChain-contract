58/// Create a new commitment-reveal round.
59pub fn create_round(round_id: u64, current_block: u32, commit_blocks: u32, reveal_blocks: u32) -> CommitRevealRound {
60    CommitRevealRound {
61        round_id,
62        commit_deadline: current_block.saturating_add(commit_blocks),
63        reveal_deadline: current_block.saturating_add(commit_blocks).saturating_add(reveal_blocks),
64        commits: Vec::new(),
65        reveals: Vec::new(),
66        final_random: None,
67        status: RandomnessStatus::Committing,
68    }
69}

71/// Add a commitment to a round during the commit phase.
72pub fn add_commit(
73    round: &mut CommitRevealRound,
74    committer: AccountId,
75    commit_hash: Hash,
76    current_block: u32,
77) -> Result<(), CryptoError> {
78    if round.status != RandomnessStatus::Committing || current_block > round.commit_deadline {
79        return Err(CryptoError::InvalidRandomnessPhase);
80    }
81    // Prevent duplicate commits from the same account
82    if round.commits.iter().any(|c| c.committer == committer) {
83        return Err(CryptoError::InvalidRandomnessPhase);
84    }
85    round.commits.push(CommitEntry {
86        committer,
87        commit_hash,
88    });
89    Ok(())
90}

92/// Transition round from committing to revealing phase.
93pub fn start_reveal_phase(round: &mut CommitRevealRound, current_block: u32) -> Result<(), CryptoError> {
94    if round.status != RandomnessStatus::Committing {
95        return Err(CryptoError::InvalidRandomnessPhase);
96    }
97    if current_block <= round.commit_deadline {
98        return Err(CryptoError::InvalidRandomnessPhase);
99    }
100    if (round.commits.len() as u32) < MIN_RANDOMNESS_PARTICIPANTS {
101        round.status = RandomnessStatus::Failed;
102        return Err(CryptoError::InsufficientReveals);
103    }
104    round.status = RandomnessStatus::Revealing;
105    Ok(())
106}

108/// Reveal a secret during the reveal phase. Verifies against the commitment.
109pub fn add_reveal(
110    round: &mut CommitRevealRound,
111    revealer: AccountId,
112    secret: [u8; 32],
113    current_block: u32,
114) -> Result<(), CryptoError> {
115    if round.status != RandomnessStatus::Revealing || current_block > round.reveal_deadline {
116        return Err(CryptoError::InvalidRandomnessPhase);
117    }
118    // Find the matching commitment
119    let commit = round
120        .commits
121        .iter()
122        .find(|c| c.committer == revealer)
123        .ok_or(CryptoError::InvalidRandomnessPhase)?;
124
125    // Verify the reveal matches the commitment
126    if !verify_commitment(&secret, &revealer, &commit.commit_hash) {
127        return Err(CryptoError::CommitMismatch);
128    }
129
130    // Prevent duplicate reveals
131    if round.reveals.iter().any(|r| r.revealer == revealer) {
132        return Err(CryptoError::InvalidRandomnessPhase);
133    }
134
135    round.reveals.push(RevealEntry { revealer, secret });
136    Ok(())
137}

139/// Finalize the round and compute the random value from all revealed secrets.
140pub fn finalize_round(round: &mut CommitRevealRound, current_block: u32) -> Result<Hash, CryptoError> {
141    if round.status != RandomnessStatus::Revealing {
142        return Err(CryptoError::InvalidRandomnessPhase);
143    }
144    if current_block <= round.reveal_deadline {
145        // Can only finalize after reveal deadline to prevent early finalization attacks
146        return Err(CryptoError::InvalidRandomnessPhase);
147    }
148    if (round.reveals.len() as u32) < MIN_RANDOMNESS_PARTICIPANTS {
149        round.status = RandomnessStatus::Failed;
150        return Err(CryptoError::InsufficientReveals);
151    }
152
153    let secrets: Vec<[u8; 32]> = round.reveals.iter().map(|r| r.secret).collect();
154    let random = finalize_randomness(&secrets);
155    round.final_random = Some(random);
156    round.status = RandomnessStatus::Finalized;
157    Ok(random)
158}
