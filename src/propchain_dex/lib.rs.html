255        pub fn create_pool(
256            &mut self,
257            base_token: TokenId,
258            quote_token: TokenId,
259            fee_bips: u32,
260            initial_base: u128,
261            initial_quote: u128,
262        ) -> Result<u64, Error> {
263            non_reentrant!(self, {
264                self.ensure_admin_or_pair_creator()?;
265                if base_token == quote_token
266                    || initial_base == 0
267                    || initial_quote == 0
268                    || fee_bips >= 1_000
269                {
270                    return Err(Error::InvalidPair);
271                }
272
273                let key = ordered_pair(base_token, quote_token);
274                if self.pair_lookup.get(key).unwrap_or(0) != 0 {
275                    return Err(Error::InvalidPair);
276                }
277
278                self.pair_counter += 1;
279                let pair_id = self.pair_counter;
280                let last_price = initial_quote
281                    .saturating_mul(BIPS_DENOMINATOR)
282                    .checked_div(initial_base)
283                    .unwrap_or(0);
284                let minted = integer_sqrt(initial_base.saturating_mul(initial_quote));
285                let pool = LiquidityPool {
286                    pair_id,
287                    base_token,
288                    quote_token,
289                    reserve_base: initial_base,
290                    reserve_quote: initial_quote,
291                    total_lp_shares: minted,
292                    fee_bips,
293                    reward_index: 0,
294                    cumulative_volume: 0,
295                    last_price,
296                    is_active: true,
297                };
298                self.pools.insert(pair_id, &pool);
299                self.pair_lookup.insert(key, &pair_id);
300                self.positions.insert(
301                    (pair_id, self.env().caller()),
302                    &LiquidityPosition {
303                        lp_shares: minted,
304                        reward_debt: 0,
305                        provided_base: initial_base,
306                        provided_quote: initial_quote,
307                        pending_rewards: 0,
308                    },
309                );
310                self.analytics.insert(
311                    pair_id,
312                    &PairAnalytics {
313                        pair_id,
314                        last_price,
315                        twap_price: last_price,
316                        reference_price: last_price,
317                        cumulative_volume: 0,
318                        trade_count: 0,
319                        best_bid: 0,
320                        best_ask: 0,
321                        volatility_bips: 0,
322                        last_updated: self.env().block_timestamp(),
323                        high_24h: last_price,
324                        low_24h: last_price,
325                        volume_24h: 0,
326                        trade_count_24h: 0,
327                    },
328                );
329                self.last_reward_block
330                    .insert(pair_id, &u64::from(self.env().block_number()));
331
332                self.env().emit_event(PoolCreated {
333                    pair_id,
334                    base_token,
335                    quote_token,
336                });
337
338                Ok(pair_id)
339            })
340        }

342        #[ink(message)]
343        pub fn add_liquidity(
344            &mut self,
345            pair_id: u64,
346            amount_base: u128,
347            amount_quote: u128,
348        ) -> Result<u128, Error> {
349            non_reentrant!(self, {
350                if amount_base == 0 || amount_quote == 0 {
351                    return Err(Error::InvalidPair);
352                }
353                self.accrue_rewards(pair_id)?;
354                let mut pool = self.pool(pair_id)?;
355                let minted_shares = if pool.total_lp_shares == 0 {
356                    integer_sqrt(amount_base.saturating_mul(amount_quote))
357                } else {
358                    let base_shares = amount_base
359                        .saturating_mul(pool.total_lp_shares)
360                        .checked_div(pool.reserve_base)
361                        .unwrap_or(0);
362                    let quote_shares = amount_quote
363                        .saturating_mul(pool.total_lp_shares)
364                        .checked_div(pool.reserve_quote)
365                        .unwrap_or(0);
366                    core::cmp::min(base_shares, quote_shares)
367                };
368
369                pool.reserve_base = pool.reserve_base.saturating_add(amount_base);
370                pool.reserve_quote = pool.reserve_quote.saturating_add(amount_quote);
371                pool.total_lp_shares = pool.total_lp_shares.saturating_add(minted_shares);
372                self.update_pool_price(&mut pool);
373                self.pools.insert(pair_id, &pool);
374
375                let caller = self.env().caller();
376                let mut position = self.position(pair_id, caller);
377                let accrued = pending_from_indices(
378                    position.lp_shares,
379                    pool.reward_index,
380                    position.reward_debt,
381                );
382                position.pending_rewards = position.pending_rewards.saturating_add(accrued);
383                position.reward_debt = scaled_reward_debt(
384                    position.lp_shares.saturating_add(minted_shares),
385                    pool.reward_index,
386                );
387                position.lp_shares = position.lp_shares.saturating_add(minted_shares);
388                position.provided_base = position.provided_base.saturating_add(amount_base);
389                position.provided_quote = position.provided_quote.saturating_add(amount_quote);
390                self.positions.insert((pair_id, caller), &position);
391
392                let mut analytics = self.analytics_for(pair_id);
393                analytics.last_updated = self.env().block_timestamp();
394                self.analytics.insert(pair_id, &analytics);
395
396                self.env().emit_event(LiquidityAdded {
397                    pair_id,
398                    provider: caller,
399                    minted_shares,
400                });
401
402                Ok(minted_shares)
403            })
404        }

406        #[ink(message)]
407        pub fn remove_liquidity(
408            &mut self,
409            pair_id: u64,
410            shares: u128,
411        ) -> Result<(u128, u128), Error> {
412            non_reentrant!(self, {
413                if shares == 0 {
414                    return Err(Error::InvalidPair);
415                }
416                self.accrue_rewards(pair_id)?;
417                let mut pool = self.pool(pair_id)?;
418                let caller = self.env().caller();
419                let mut position = self.position(pair_id, caller);
420                if shares > position.lp_shares {
421                    return Err(Error::InsufficientLiquidity);
422                }
423
424                let base_out = shares
425                    .saturating_mul(pool.reserve_base)
426                    .checked_div(pool.total_lp_shares)
427                    .unwrap_or(0);
428                let quote_out = shares
429                    .saturating_mul(pool.reserve_quote)
430                    .checked_div(pool.total_lp_shares)
431                    .unwrap_or(0);
432                pool.reserve_base = pool.reserve_base.saturating_sub(base_out);
433                pool.reserve_quote = pool.reserve_quote.saturating_sub(quote_out);
434                pool.total_lp_shares = pool.total_lp_shares.saturating_sub(shares);
435                self.update_pool_price(&mut pool);
436                self.pools.insert(pair_id, &pool);
437
438                let accrued = pending_from_indices(
439                    position.lp_shares,
440                    pool.reward_index,
441                    position.reward_debt,
442                );
443                position.pending_rewards = position.pending_rewards.saturating_add(accrued);
444                position.lp_shares = position.lp_shares.saturating_sub(shares);
445                position.reward_debt = scaled_reward_debt(position.lp_shares, pool.reward_index);
446                self.positions.insert((pair_id, caller), &position);
447
448                Ok((base_out, quote_out))
449            })

477        #[allow(clippy::too_many_arguments)]
478        pub fn place_order(
479            &mut self,
480            pair_id: u64,
481            side: OrderSide,
482            order_type: OrderType,
483            time_in_force: TimeInForce,
484            price: u128,
485            amount: u128,
486            trigger_price: Option<u128>,
487            twap_interval: Option<u64>,
488            reduce_only: bool,
489        ) -> Result<u64, Error> {
490            non_reentrant!(self, {
491                if amount == 0 {
492                    return Err(Error::InvalidOrder);
493                }
494                let _ = self.pool(pair_id)?;
495                if matches!(
496                    order_type,
497                    OrderType::Limit | OrderType::StopLoss | OrderType::TakeProfit
498                ) && price == 0
499                {
500                    return Err(Error::InvalidOrder);
501                }
502
503                self.order_counter += 1;
504                let now = self.env().block_timestamp();
505                let order_id = self.order_counter;
506                let order = TradingOrder {
507                    order_id,
508                    pair_id,
509                    trader: self.env().caller(),
510                    side,
511                    order_type,
512                    time_in_force,
513                    price,
514                    amount,
515                    remaining_amount: amount,
516                    trigger_price,
517                    twap_interval,
518                    reduce_only,
519                    status: OrderStatus::Open,
520                    created_at: now,
521                    updated_at: now,
522                };
523                self.orders.insert(order_id, &order);
524                let count = self.order_book_count.get(pair_id).unwrap_or(0);
525                self.order_book.insert((pair_id, count), &order_id);
526                self.order_book_count.insert(pair_id, &(count + 1));
527
528                self.refresh_best_quotes(pair_id);
529
530                self.env().emit_event(OrderPlaced {
531                    order_id,
532                    pair_id,
533                    trader: self.env().caller(),
534                });
535
536                if matches!(
537                    time_in_force,
538                    TimeInForce::ImmediateOrCancel | TimeInForce::FillOrKill
539                ) || matches!(order_type, OrderType::Market)
540                {
541                    self.execute_order(order_id, amount)?;
542                }
543
544                Ok(order_id)
545            })
546        }

559        fn execute_order_core(
560            &mut self,
561            order_id: u64,
562            requested_amount: u128,
563        ) -> Result<u128, Error> {
564            let mut order = self.order(order_id)?;
565            if !matches!(
566                order.status,
567                OrderStatus::Open | OrderStatus::PartiallyFilled | OrderStatus::Triggered
568            ) {
569                return Err(Error::OrderNotExecutable);
570            }
571
572            let executable = self.is_order_executable(&order)?;
573            if !executable {
574                return Err(Error::OrderNotExecutable);
575            }
576
577            let fill_amount = core::cmp::min(requested_amount, order.remaining_amount);
578            if fill_amount == 0 {
579                return Err(Error::InvalidOrder);
580            }
581
582            let pair_id = order.pair_id;
583            let output = match order.side {
584                OrderSide::Sell => self.swap(pair_id, OrderSide::Sell, fill_amount, 0)?,
585                OrderSide::Buy => self.swap(pair_id, OrderSide::Buy, fill_amount, 0)?,
586            };
587
588            order.remaining_amount = order.remaining_amount.saturating_sub(fill_amount);
589            order.updated_at = self.env().block_timestamp();
590            order.status = if order.remaining_amount == 0 {
591                OrderStatus::Filled
592            } else {
593                OrderStatus::PartiallyFilled
594            };
595            self.orders.insert(order_id, &order);
596            self.refresh_best_quotes(pair_id);
597
598            Ok(output)
599        }

601        #[ink(message)]
602        pub fn match_orders(
603            &mut self,
604            maker_order_id: u64,
605            taker_order_id: u64,
606            amount: u128,
607        ) -> Result<u128, Error> {
608            non_reentrant!(self, {
609                let mut maker = self.order(maker_order_id)?;
610                let mut taker = self.order(taker_order_id)?;
611                if maker.pair_id != taker.pair_id || maker.side == taker.side {
612                    return Err(Error::InvalidOrder);
613                }
614
615                let fill_amount = core::cmp::min(
616                    amount,
617                    core::cmp::min(maker.remaining_amount, taker.remaining_amount),
618                );
619                if fill_amount == 0 {
620                    return Err(Error::InvalidOrder);
621                }
622
623                let execution_price = if maker.price > 0 {
624                    maker.price
625                } else {
626                    taker.price
627                };
628                let notional = fill_amount
629                    .saturating_mul(execution_price)
630                    .checked_div(BIPS_DENOMINATOR)
631                    .unwrap_or(0);
632
633                maker.remaining_amount = maker.remaining_amount.saturating_sub(fill_amount);
634                taker.remaining_amount = taker.remaining_amount.saturating_sub(fill_amount);
635                maker.status = if maker.remaining_amount == 0 {
636                    OrderStatus::Filled
637                } else {
638                    OrderStatus::PartiallyFilled
639                };
640                taker.status = if taker.remaining_amount == 0 {
641                    OrderStatus::Filled
642                } else {
643                    OrderStatus::PartiallyFilled
644                };
645                maker.updated_at = self.env().block_timestamp();
646                taker.updated_at = maker.updated_at;
647                self.orders.insert(maker_order_id, &maker);
648                self.orders.insert(taker_order_id, &taker);
649
650                let mut analytics = self.analytics_for(maker.pair_id);
651                let prev = analytics.last_price;
652                analytics.last_price = execution_price;
653                analytics.reference_price =
654                    weighted_average(execution_price, analytics.twap_price, 7, 3);
655                analytics.twap_price =
656                    weighted_average(execution_price, analytics.twap_price, 1, 1);
657                analytics.cumulative_volume = analytics.cumulative_volume.saturating_add(notional);
658                analytics.trade_count = analytics.trade_count.saturating_add(1);
659                analytics.volatility_bips = volatility_bips(prev, execution_price);
660                analytics.last_updated = self.env().block_timestamp();
661                self.analytics.insert(maker.pair_id, &analytics);
662                self.refresh_best_quotes(maker.pair_id);
663
664                Ok(notional)
665            })
666        }

868        #[ink(message)]
869        pub fn claim_competition_reward(&mut self, competition_id: u64) -> Result<u128, Error> {
870            let competition = self
871                .trading_competitions
872                .get(competition_id)
873                .ok_or(Error::InvalidRequest)?;
874            let current_block = u64::from(self.env().block_number());
875            if current_block <= competition.end_block {
876                return Err(Error::InvalidRequest);
877            }
878            let caller = self.env().caller();
879            if self
880                .competition_claimed
881                .get((competition_id, caller))
882                .unwrap_or(false)
883            {
884                return Err(Error::InvalidRequest);
885            }
886            let score = self
887                .competition_scores
888                .get((competition_id, caller))
889                .unwrap_or(0);
890            if score == 0 {
891                return Err(Error::InvalidRequest);
892            }
893
894            let participants = self
895                .competition_participants
896                .get(competition_id)
897                .unwrap_or_else(Vec::new);
898            let mut total_score = 0u128;
899            for participant in participants.iter() {
900                total_score = total_score.saturating_add(
901                    self.competition_scores
902                        .get((competition_id, *participant))
903                        .unwrap_or(0),
904                );
905            }
906            if total_score == 0 {
907                return Err(Error::InvalidRequest);
908            }
909
910            let reward = competition
911                .reward_amount
912                .saturating_mul(score)
913                .checked_div(total_score)
914                .unwrap_or(0);
915
916            if reward == 0 {
917                return Err(Error::InvalidRequest);
918            }
919
920            self.competition_claimed
921                .insert((competition_id, caller), &true);
922            let balance = self.governance_balances.get(caller).unwrap_or(0);
923            self.governance_balances
924                .insert(caller, &balance.saturating_add(reward));
925            self.governance_config.total_supply =
926                self.governance_config.total_supply.saturating_add(reward);
927            self.env().emit_event(CompetitionRewardClaimed {
928                competition_id,
929                trader: caller,
930                reward_amount: reward,
931            });
932            Ok(reward)
933        }

935        fn update_trade_competition_score(
936            &mut self,
937            pair_id: u64,
938            trader: AccountId,
939            volume: u128,
940        ) {
941            let current_block = u64::from(self.env().block_number());
942            for competition_id in 1..=self.trade_competition_counter {
943                if let Some(competition) = self.trading_competitions.get(competition_id) {
944                    if !competition.active
945                        || current_block < competition.start_block
946                        || current_block > competition.end_block
947                    {
948                        continue;
949                    }
950                    if let Some(id) = competition.pair_id {
951                        if id != pair_id {
952                            continue;
953                        }
954                    }
955                    if volume < competition.min_trade_volume {
956                        continue;
957                    }
958
959                    let key = (competition_id, trader);
960                    let prev_score = self.competition_scores.get(key).unwrap_or(0);
961                    let next_score = prev_score.saturating_add(volume);
962                    self.competition_scores.insert(key, &next_score);
963
964                    let mut participants = self
965                        .competition_participants
966                        .get(competition_id)
967                        .unwrap_or_else(Vec::new);
968                    if !participants.contains(&trader) {
969                        participants.push(trader);
970                        self.competition_participants
971                            .insert(competition_id, &participants);
972                    }
973
974                    self.env().emit_event(CompetitionScoreUpdated {
975                        competition_id,
976                        trader,
977                        score: next_score,
978                    });
979                }
980            }
981        }

1742        /// Calculate the expected price impact for a given trade amount.
1743        /// Returns the price impact in basis points (bips) and the expected output amount.
1744        /// This allows users to check the impact before executing a trade.
1745        #[ink(message)]
1746        pub fn calculate_price_impact(
1747            &self,
1748            pair_id: u64,
1749            side: OrderSide,
1750            amount_in: u128,
1751        ) -> Result<(u32, u128), Error> {
1752            if amount_in == 0 {
1753                return Err(Error::InvalidOrder);
1754            }
1755            let pool = self.pool(pair_id)?;
1756            let fee_adjusted_in = amount_in
1757                .saturating_mul(BIPS_DENOMINATOR.saturating_sub(pool.fee_bips as u128))
1758                .checked_div(BIPS_DENOMINATOR)
1759                .unwrap_or(0);
1760
1761            let (reserve_in, reserve_out) = match side {
1762                OrderSide::Sell => (pool.reserve_base, pool.reserve_quote),
1763                OrderSide::Buy => (pool.reserve_quote, pool.reserve_base),
1764            };
1765            if reserve_in == 0 || reserve_out == 0 {
1766                return Err(Error::InsufficientLiquidity);
1767            }
1768
1769            let amount_out = fee_adjusted_in
1770                .saturating_mul(reserve_out)
1771                .checked_div(reserve_in.saturating_add(fee_adjusted_in))
1772                .unwrap_or(0);
1773
1774            let price_before = reserve_out
1775                .saturating_mul(BIPS_DENOMINATOR)
1776                .checked_div(reserve_in)
1777                .unwrap_or(0);
1778
1779            let reserve_in_after = reserve_in.saturating_add(amount_in);
1780            let reserve_out_after = reserve_out.saturating_sub(amount_out);
1781            let price_after = if reserve_in_after > 0 {
1782                reserve_out_after
1783                    .saturating_mul(BIPS_DENOMINATOR)
1784                    .checked_div(reserve_in_after)
1785                    .unwrap_or(0)
1786            } else {
1787                0
1788            };
1789
1790            let price_impact_bips = if price_before > 0 {
1791                price_before
1792                    .abs_diff(price_after)
1793                    .saturating_mul(BIPS_DENOMINATOR)
1794                    .checked_div(price_before)
1795                    .unwrap_or(0) as u32
1796            } else {
1797                0
1798            };
1799
1800            Ok((price_impact_bips, amount_out))
1801        }
