1// Builder module for the contract factory.
2//
3// Loaded via `pub mod builder;` in `lib.rs`. The `deploy_contract` ink! message
4// calls `builder::build_contract(contract_type, init_params, salt) -> Result<BuildResult, Error>`
5// as part of the deterministic-deployment flow. In a production chain this
6// would compute a CREATE2 address from `code_hash + salt + init_params`; here
7// it returns a synthetic `BuildResult` so the factory crate compiles when
8// the deeper cross-contract instantiation plumbing is not yet wired in.
9
10use ink::prelude::vec::Vec;
11
12/// Outcome of a builder-callable contract construction. `address` would be
13/// the CREATE2-derived contract address in a real implementation.
14pub struct BuildResult {
15    pub address: ink::primitives::AccountId,
16}
17
18/// Errors specific to the builder (placeholder).
19#[derive(Debug, PartialEq, Eq, scale::Encode, scale::Decode)]
20#[cfg_attr(feature = "std", derive(scale_info::TypeInfo))]
21pub enum BuildError {
22    InvalidParams,
23}
24
25/// Stub for `build_contract` — takes the project's `ContractType` enum (a
26/// `Copy + PartialEq + Eq + scale::Encode + scale::Decode` type), ignores
27/// `_init_params` and `_salt`, and returns the zero `AccountId`.  Replace
28/// with a real CREATE2 instantiation when the factory connects to a
29/// deployer on-chain.
30pub fn build_contract(
31    _contract_type: super::contract_factory::ContractType,
32    _init_params: Vec<u8>,
33    _salt: Option<[u8; 32]>,
34) -> Result<BuildResult, BuildError> {
35    Ok(BuildResult {
36        address: ink::primitives::AccountId::from([0u8; 32]),
37    })
38}
