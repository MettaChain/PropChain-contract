379        /// Create a new identity with DID
380        #[ink(message)]
381        pub fn create_identity(
382            &mut self,
383            did: String,
384            public_key: Vec<u8>,
385            verification_method: String,
386            service_endpoint: Option<String>,
387            privacy_settings: PrivacySettings,
388        ) -> Result<(), IdentityError> {
389            let caller = self.env().caller();
390            let timestamp = self.env().block_timestamp();
391
392            // Check if identity already exists
393            if self.identities.contains(&caller) {
394                return Err(IdentityError::IdentityAlreadyExists);
395            }
396
397            // Validate DID format
398            if !self.validate_did_format(&did) {
399                return Err(IdentityError::InvalidDid);
400            }
401
402            // Create DID document
403            let did_document = DIDDocument {
404                did: did.clone(),
405                public_key,
406                verification_method,
407                service_endpoint,
408                created_at: timestamp,
409                updated_at: timestamp,
410                version: 1,
411            };
412
413            // Create social recovery config with default settings
414            let social_recovery = SocialRecoveryConfig {
415                guardians: Vec::new(),
416                threshold: 3,
417                recovery_period: 100800, // ~2 weeks in blocks (assuming 6s block time)
418                last_recovery_attempt: None,
419                is_recovery_active: false,
420                recovery_approvals: Vec::new(),
421            };
422
423            // Create identity
424            let identity = Identity {
425                account_id: caller,
426                did_document,
427                reputation_score: 500, // Start with neutral reputation
428                verification_level: VerificationLevel::None,
429                trust_score: 50,
430                is_verified: false,
431                verified_at: None,
432                verification_expires: None,
433                social_recovery,
434                privacy_settings,
435                created_at: timestamp,
436                last_activity: timestamp,
437            };
438
439            // Store identity
440            self.identities.insert(&caller, &identity);
441            self.did_to_account.insert(&did, &caller);
442
443            // Initialize reputation metrics
444            let reputation_metrics = ReputationMetrics {
445                total_transactions: 0,
446                successful_transactions: 0,
447                failed_transactions: 0,
448                dispute_count: 0,
449                dispute_resolved_count: 0,
450                average_transaction_value: 0,
451                total_value_transacted: 0,
452                last_updated: timestamp,
453                reputation_score: 500,
454            };
455            self.reputation_metrics.insert(&caller, &reputation_metrics);
456
457            // Emit event
458            self.env().emit_event(IdentityCreated {
459                account: caller,
460                did,
461                timestamp,
462            });
463
464            Ok(())
465        }

467        /// Verify identity (verifier only)
468        #[ink(message)]
469        pub fn verify_identity(
470            &mut self,
471            target_account: AccountId,
472            verification_level: VerificationLevel,
473            expires_in_days: Option<u64>,
474        ) -> Result<(), IdentityError> {
475            let caller = self.env().caller();
476            let timestamp = self.env().block_timestamp();
477
478            // Check if caller is authorized verifier
479            if !self.is_authorized_verifier(caller) {
480                return Err(IdentityError::Unauthorized);
481            }
482
483            // Get identity
484            let mut identity = self
485                .identities
486                .get(&target_account)
487                .ok_or(IdentityError::IdentityNotFound)?;
488
489            // Update verification
490            identity.verification_level = verification_level;
491            identity.is_verified = true;
492            identity.verified_at = Some(timestamp);
493            identity.verification_expires = expires_in_days.map(|days| timestamp + days * 86400);
494            identity.last_activity = timestamp;
495
496            // Update trust score based on verification level
497            identity.trust_score = match verification_level {
498                VerificationLevel::None => 0,
499                VerificationLevel::Basic => 60,
500                VerificationLevel::Standard => 75,
501                VerificationLevel::Enhanced => 90,
502                VerificationLevel::Premium => 100,
503            };
504
505            // Store updated identity
506            self.identities.insert(&target_account, &identity);
507
508            // Emit event
509            self.env().emit_event(IdentityVerified {
510                account: target_account,
511                verification_level,
512                verified_by: caller,
513                timestamp,
514            });
515
516            Ok(())
517        }

519        /// Update reputation based on transaction
520        #[ink(message)]
521        pub fn update_reputation(
522            &mut self,
523            target_account: AccountId,
524            transaction_successful: bool,
525            transaction_value: u128,
526        ) -> Result<(), IdentityError> {
527            let caller = self.env().caller();
528            let timestamp = self.env().block_timestamp();
529
530            // Only authorized contracts can update reputation
531            if !self.is_authorized_verifier(caller) {
532                return Err(IdentityError::Unauthorized);
533            }
534
535            // Get and update reputation metrics
536            let mut metrics = self
537                .reputation_metrics
538                .get(&target_account)
539                .unwrap_or(ReputationMetrics {
540                    total_transactions: 0,
541                    successful_transactions: 0,
542                    failed_transactions: 0,
543                    dispute_count: 0,
544                    dispute_resolved_count: 0,
545                    average_transaction_value: 0,
546                    total_value_transacted: 0,
547                    last_updated: timestamp,
548                    reputation_score: 500,
549                });
550
551            metrics.total_transactions += 1;
552            metrics.total_value_transacted += transaction_value;
553            metrics.average_transaction_value =
554                metrics.total_value_transacted / metrics.total_transactions as u128;
555
556            if transaction_successful {
557                metrics.successful_transactions += 1;
558                // Increase reputation for successful transactions
559                metrics.reputation_score = (metrics.reputation_score + 5).min(1000);
560            } else {
561                metrics.failed_transactions += 1;
562                // Decrease reputation for failed transactions
563                metrics.reputation_score = metrics.reputation_score.saturating_sub(10);
564            }
565
566            metrics.last_updated = timestamp;
567
568            // Update identity reputation score
569            if let Some(mut identity) = self.identities.get(&target_account) {
570                let old_score = identity.reputation_score;
571                identity.reputation_score = metrics.reputation_score;
572                identity.last_activity = timestamp;
573                self.identities.insert(&target_account, &identity);
574
575                // Emit event
576                self.env().emit_event(ReputationUpdated {
577                    account: target_account,
578                    old_score,
579                    new_score: metrics.reputation_score,
580                    timestamp,
581                });
582            }
583
584            // Store updated metrics
585            self.reputation_metrics.insert(&target_account, &metrics);
586
587            Ok(())
588        }

590        /// Get trust assessment for counterparty
591        #[ink(message)]
592        pub fn assess_trust(
593            &mut self,
594            target_account: AccountId,
595        ) -> Result<TrustAssessment, IdentityError> {
596            let caller = self.env().caller();
597            let timestamp = self.env().block_timestamp();
598
599            // Get target identity and reputation
600            let target_identity = self
601                .identities
602                .get(&target_account)
603                .ok_or(IdentityError::IdentityNotFound)?;
604            let target_metrics =
605                self.reputation_metrics
606                    .get(&target_account)
607                    .unwrap_or(ReputationMetrics {
608                        total_transactions: 0,
609                        successful_transactions: 0,
610                        failed_transactions: 0,
611                        dispute_count: 0,
612                        dispute_resolved_count: 0,
613                        average_transaction_value: 0,
614                        total_value_transacted: 0,
615                        last_updated: timestamp,
616                        reputation_score: target_identity.reputation_score,
617                    });
618
619            // Calculate trust score
620            let trust_score = self.calculate_trust_score(&target_identity, &target_metrics);
621
622            // Determine risk level based on trust score
623            let risk_level = if trust_score >= 80 {
624                RiskLevel::Low
625            } else if trust_score >= 60 {
626                RiskLevel::Medium
627            } else if trust_score >= 40 {
628                RiskLevel::High
629            } else {
630                RiskLevel::Critical
631            };
632
633            // Create trust assessment
634            let assessment = TrustAssessment {
635                target_account,
636                trust_score,
637                risk_level,
638                verification_level: target_identity.verification_level,
639                reputation_score: target_identity.reputation_score,
640                shared_transactions: target_metrics.total_transactions,
641                positive_interactions: target_metrics.successful_transactions,
642                negative_interactions: target_metrics.failed_transactions,
643                assessment_date: timestamp,
644                expires_at: timestamp + 86400 * 30, // 30 days
645            };
646
647            self.trust_assessments
648                .insert(&(caller, target_account), &assessment);
649
650            Ok(assessment)
651        }

705        /// Initiate social recovery
706        #[ink(message)]
707        pub fn initiate_recovery(
708            &mut self,
709            new_account: AccountId,
710            recovery_signature: Vec<u8>,
711        ) -> Result<(), IdentityError> {
712            let caller = self.env().caller();
713            let timestamp = self.env().block_timestamp();
714
715            // Get identity
716            let mut identity = self
717                .identities
718                .get(&caller)
719                .ok_or(IdentityError::IdentityNotFound)?;
720
721            // Check if recovery is already in progress
722            if identity.social_recovery.is_recovery_active {
723                return Err(IdentityError::RecoveryInProgress);
724            }
725
726            // Verify recovery signature
727            if !self.verify_recovery_signature(
728                &caller,
729                &new_account,
730                &recovery_signature,
731                &identity,
732            ) {
733                return Err(IdentityError::InvalidSignature);
734            }
735
736            // Start recovery process
737            identity.social_recovery.is_recovery_active = true;
738            identity.social_recovery.last_recovery_attempt = Some(timestamp);
739            identity.social_recovery.recovery_approvals = Vec::new();
740
741            // Store updated identity
742            self.identities.insert(&caller, &identity);
743
744            // Emit event
745            self.env().emit_event(RecoveryInitiated {
746                account: caller,
747                initiator: caller,
748                timestamp,
749            });
750
751            Ok(())
752        }

754        /// Approve recovery (guardian only)
755        #[ink(message)]
756        pub fn approve_recovery(
757            &mut self,
758            target_account: AccountId,
759            new_account: AccountId,
760        ) -> Result<(), IdentityError> {

762
763            // Get target identity

766                .get(&target_account)

769            // Check if caller is a guardian
770            if !identity.social_recovery.guardians.contains(&caller) {
771                return Err(IdentityError::Unauthorized);
772            }
773
774            // Check if recovery is active
775            if !identity.social_recovery.is_recovery_active {
776                return Err(IdentityError::RecoveryNotActive);
777            }
778
779            // Add approval
780            if !identity
781                .social_recovery
782                .recovery_approvals
783                .contains(&caller)
784            {
785                identity.social_recovery.recovery_approvals.push(caller);
786            }
787
788            // Check if threshold is met
789            if identity.social_recovery.recovery_approvals.len()
790                >= identity.social_recovery.threshold as usize
791            {
792                // Complete recovery
793                self.complete_recovery(target_account, new_account)?;
794            } else {
795                // Store updated identity
796                self.identities.insert(&target_account, &identity);
797            }
798
799            Ok(())
800        }

802        /// Complete identity recovery
803        fn complete_recovery(
804            &mut self,
805            old_account: AccountId,
806            new_account: AccountId,
807        ) -> Result<(), IdentityError> {
808            let _timestamp = self.env().block_timestamp();
809
810            // Get old identity
811            let mut identity = self
812                .identities
813                .get(&old_account)
814                .ok_or(IdentityError::IdentityNotFound)?;
815
816            // Update account ID
817            identity.account_id = new_account;
818            identity.social_recovery.is_recovery_active = false;
819            identity.social_recovery.recovery_approvals = Vec::new();
820            identity.last_activity = _timestamp;
821
822            // Remove old identity mapping
823            self.identities.remove(&old_account);
824
825            // Add new identity mapping
826            self.identities.insert(&new_account, &identity);
827            self.did_to_account
828                .insert(&identity.did_document.did, &new_account);
829
830            // Update reputation metrics mapping
831            if let Some(metrics) = self.reputation_metrics.get(&old_account) {
832                self.reputation_metrics.remove(&old_account);
833                self.reputation_metrics.insert(&new_account, &metrics);
834            }
835
836            // Emit event
837            self.env().emit_event(RecoveryCompleted {
838                account: old_account,
839                new_account,
840                timestamp: _timestamp,
841            });
842
843            Ok(())
844        }

846        /// Privacy-preserving identity verification using zero-knowledge proofs
847        #[ink(message)]
848        pub fn verify_privacy_preserving(
849            &mut self,
850            proof: Vec<u8>,
851            public_inputs: Vec<u8>,
852            verification_type: String,
853        ) -> Result<bool, IdentityError> {
854            let caller = self.env().caller();
855            let _timestamp = self.env().block_timestamp();
856
857            // Get identity
858            let identity = self
859                .identities
860                .get(&caller)
861                .ok_or(IdentityError::IdentityNotFound)?;
862
863            // Check if privacy settings allow this verification
864            if !identity.privacy_settings.zero_knowledge_proof {
865                return Err(IdentityError::PrivacyVerificationFailed);
866            }
867
868            // Verify zero-knowledge proof (simplified verification)
869            let is_valid =
870                self.verify_zero_knowledge_proof(&proof, &public_inputs, &verification_type);
871
872            if is_valid {
873                // Update privacy nonce for replay protection
874                let current_nonce = self.privacy_nonces.get(&caller).unwrap_or(0);
875                self.privacy_nonces.insert(&caller, &(current_nonce + 1));
876
877                // Update last activity
878                let mut updated_identity = identity;
879                updated_identity.last_activity = _timestamp;
880                self.identities.insert(&caller, &updated_identity);
881            }
882
883            Ok(is_valid)
884        }

938        fn calculate_trust_score(&self, identity: &Identity, metrics: &ReputationMetrics) -> u32 {
939            let base_score = identity.trust_score;
940            let reputation_factor = identity.reputation_score;
941            let verification_bonus = match identity.verification_level {
942                VerificationLevel::None => 0,
943                VerificationLevel::Basic => 10,
944                VerificationLevel::Standard => 20,
945                VerificationLevel::Enhanced => 30,
946                VerificationLevel::Premium => 40,
947            };
948
949            // Calculate success rate
950            #[allow(clippy::manual_checked_ops)]
951            let success_rate = if metrics.total_transactions > 0 {
952                (metrics.successful_transactions * 100) / metrics.total_transactions
953            } else {
954                50 // Default for no history
955            };
956
957            // Weighted calculation with proper type casting
958            ((base_score as u64 * 40)
959                + (reputation_factor as u64 / 10 * 30)
960                + (verification_bonus as u64 * 20)
961                + (success_rate * 10)) as u32
962                / 100
963        }
